001package com.github.gwtbootstrap.datepicker.client.ui.util;
002
003import com.github.gwtbootstrap.client.ui.resources.JavaScriptInjector;
004import com.github.gwtbootstrap.datepicker.client.ui.resources.DatepickerResourceInjector;
005import com.github.gwtbootstrap.datepicker.client.ui.resources.Resources;
006import com.google.gwt.resources.client.TextResource;
007
008import java.util.ArrayList;
009import java.util.List;
010
011/**
012 * A utility class to get the User's Browser Locale.
013 *
014 * @author Carlos A Becker
015 * @since 2.0.3.0
016 */
017public class LocaleUtil {
018
019    private static String locale = null;
020    private static String LANGUAGE = null;
021
022    private static List<String> loaded = new ArrayList<String>();
023
024    public static String getLanguage() {
025        if (LANGUAGE == null) {
026            setupLocale();
027        }
028        return LANGUAGE;
029    }
030
031    /**
032     * Get the string of locale based on user's browser configuration.
033     *
034     * @return
035     */
036    public static String getLocale() {
037        if (locale == null) {
038            locale = getBrowserLocale();
039        }
040        return locale;
041    }
042
043    public static TextResource getLocaleJsResource() {
044        String locale = getLocale();
045        if (locale == null) {
046            return null;
047        }
048        return setupLocale();
049    }
050
051    private static final native String getBrowserLocale() /*-{
052        return $wnd.navigator.userLanguage || $wnd.navigator.language;
053    }-*/;
054
055
056    public static final void forceLocale(String locale_) {
057        locale = locale_;
058        TextResource t = setupLocale();
059        if (!loaded.contains(locale) && t != null) {
060            JavaScriptInjector.inject(t.getText());
061        }
062    }
063
064    private static TextResource setupLocale() {
065        Resources r = Resources.RESOURCES;
066        TextResource tr = null;
067
068        /*
069           Script used to gen the basic if-else block:
070           for a in `ls`; do echo "else if(locale.equals(\"`echo $a | cut -f2 -d.`\")) { tr = r.`echo $a | cut -f2 -d.`(); LANGUAGE = \"`echo $a | cut -f2 -d.`\"; }"; done
071        */
072
073
074        if (locale.equalsIgnoreCase("pt-br") || locale.contains("pt")) {
075            tr = r.br();
076            LANGUAGE = "br";
077        } else if (locale.equals("da")) {
078            tr = r.da();
079            LANGUAGE = "da";
080        } else if (locale.equals("de")) {
081            tr = r.de();
082            LANGUAGE = "de";
083        } else if (locale.equals("es")) {
084            tr = r.es();
085            LANGUAGE = "es";
086        } else if (locale.equals("fi")) {
087            tr = r.fi();
088            LANGUAGE = "fi";
089        } else if (locale.equals("fr")) {
090            tr = r.fr();
091            LANGUAGE = "fr";
092        } else if (locale.equals("is")) {
093            tr = r.is();
094            LANGUAGE = "is";
095        } else if (locale.equals("it")) {
096            tr = r.it();
097            LANGUAGE = "it";
098        } else if (locale.equals("lv")) {
099            tr = r.lv();
100            LANGUAGE = "lv";
101        } else if (locale.equals("nb")) {
102            tr = r.nb();
103            LANGUAGE = "nb";
104        } else if (locale.equals("nl")) {
105            tr = r.nl();
106            LANGUAGE = "nl";
107        } else if (locale.equals("pl")) {
108            tr = r.pl();
109            LANGUAGE = "pl";
110        } else if (locale.equals("ru")) {
111            tr = r.ru();
112            LANGUAGE = "ru";
113        } else if (locale.equals("sv")) {
114            tr = r.sv();
115            LANGUAGE = "sv";
116        } else if (locale.equals("th")) {
117            tr = r.th();
118            LANGUAGE = "th";
119        } else if (locale.equals("tr")) {
120            tr = r.tr();
121            LANGUAGE = "tr";
122        } else if (locale.equals("zh-CN")) {
123            tr = r.zh_CN();
124            LANGUAGE = "zh-CN";
125        } else {
126            tr = null;
127            LANGUAGE = "en";
128        }
129
130        loaded.add(LANGUAGE);
131
132        return tr;
133    }
134}
