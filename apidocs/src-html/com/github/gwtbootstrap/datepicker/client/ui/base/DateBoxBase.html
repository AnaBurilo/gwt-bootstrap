005    import com.github.gwtbootstrap.client.ui.base.HasVisibleHandlers;

017    import java.util.Date;

019    /**
020     * Base DatePicker component.
021     *
022     * @author Carlos Alexandro Becker
023     * @since 2.0.3.0
024     */
025    public class DateBoxBase extends Widget implements HasValue<Date>, HasValueChangeHandlers<Date>, HasVisibility,
026            HasVisibleHandlers, HasAllDatePickerHandlers {
027    
028        private final TextBox box;
029        private String format;
030        private String language;
031        private DateTimeFormat dtf;
032    
033        public DateBoxBase() {
034            this.box = new TextBox();
035            setElement(box.getElement());
036            setFormat("mm/dd/yyyy");
037            this.language = LocaleUtil.getLanguage();
038            setValue(new Date());
039        }
040    
041        /**
042         * {@inheritDoc}
043         */
044        @Override
045        public void setFormat(String format) {
046            this.format = format;
047            Date oldValue = getValue();
048            this.dtf = DateTimeFormat.getFormat(format.replaceAll("mm", "MM"));
049            if (oldValue != null) {
050                setValue(oldValue);
051            }
052        }
053    
054        public void setLanguage(String language) {
055            this.language = language;
056            LocaleUtil.forceLocale(language);
057        }
058    
059        /**
060         * Returns the internal instance of textbox element. Use only if know what you are doing.
061         *
062         * @return internal textbox intance.
063         */
064        protected TextBox getBox() {
065            return box;
066        }
067    
068        /**
069         * {@inheritDoc}
070         */
071        @Override
072        public Date getValue() {
073            return dtf != null ? dtf.parse(box.getValue()) : null;
074        }
075    
076        /**
077         * {@inheritDoc}
078         */
079        @Override
080        public void setValue(Date value) {
081            setValue(value, false);
082        }
083    
084        /**
085         * {@inheritDoc}
086         */
087        @Override
088        public void setValue(Date value, boolean fireEvents) {
089            box.setValue(dtf.format(value));
090            if (fireEvents) {
091                ValueChangeEvent.fire(this, value);
092            }
093        }
094    
095        /**
096         * {@inheritDoc}
097         */
098        @Override
099        public HandlerRegistration addValueChangeHandler(ValueChangeHandler<Date> dateValueChangeHandler) {
100            return addHandler(dateValueChangeHandler, ValueChangeEvent.getType());
101        }
102    
103        /**
104         * {@inheritDoc}
105         */
106        @Override
107        protected void onLoad() {
108            super.onLoad();
109            configure();
110        }
111    
112        /**
113         * Configure the elements for a specific widget.
114         * Use only if you know what you are doing.
115         *
116         * @param w: the widget to configure.
117         */
118        protected void configure(Widget w) {
119            w.getElement().setAttribute("data-date-format", format);
120            w.getElement().setAttribute("data-date-language", language);
121            configure(w.getElement());
122        }
123    
124        /**
125         * dateChange event handler.
126         */
127        public void onChange() {
128            ValueChangeEvent.fire(this, getValue());
129        }
130    
131        /**
132         * configure this datepicker.
133         */
134        protected void configure() {
135            configure(this);
136        }
137    
138        /**
139         * call jquery datepicker plugin in a element.
140         *
141         * @param e: Element that will be transformed in a datepicker.
142         */
143        protected native void configure(Element e) /*-{
144            var that = this;
145            $wnd.jQuery(e).datepicker();

147    //            that.@com.github.gwtbootstrap.datepicker.client.ui.base.DateBoxBase::onChange()();
148    //        });
149    //        $wnd.jQuery(e).on("show", function () {
150    //            that.@com.github.gwtbootstrap.datepicker.client.ui.base.DateBoxBase::show()();
151    //        });
152    //        $wnd.jQuery(e).on("hide", function () {
153    //            that.@com.github.gwtbootstrap.datepicker.client.ui.base.DateBoxBase::hide()();
154    //        });
155        }-*/;
156    
157        private native void execute(Element e, String cmd) /*-{
158            $wnd.jQuery(e).datepicker(cmd);
159        }-*/;
160    
161        private void execute(String cmd) {
162            execute(getElement(), cmd);
163        }
164    
165        /**
166         * {@inheritDoc}
167         */
168        @Override
169        public void show() {
170            execute("show");
171        }
172    
173        /**
174         * {@inheritDoc}
175         */
176        @Override
177        public void hide() {
178            execute("hide");
179        }
180    
181        /**
182         * {@inheritDoc}
183         */
184        @Override
185        public void toggle() {
186            // TODO toggle.
187        }
188    
189        /**
190         * {@inheritDoc}
191         */
192        @Override
193        public HandlerRegistration addHideHandler(HideHandler handler) {
194            return addHandler(handler, HideEvent.getType());
195        }
196    
197        /**
198         * {@inheritDoc}
199         */
200        @Override
201        public HandlerRegistration addHiddenHandler(HiddenHandler handler) {
202            return addHandler(handler, HiddenEvent.getType());
203        }
204    
205        /**
206         * {@inheritDoc}
207         */
208        @Override
209        public HandlerRegistration addShowHandler(ShowHandler handler) {
210            return addHandler(handler, ShowEvent.getType());
211        }
212    
213        /**
214         * {@inheritDoc}
215         */
216        @Override
217        public HandlerRegistration addShownHandler(ShownHandler handler) {
218            return addHandler(handler, ShownEvent.getType());
219        }
220    
221        /**
222         * {@inheritDoc}
223         */
224        @Override
225        public void setWeekStart(int start) {
226            getElement().setAttribute("data-date-weekstart", start + "");
227        }
228    
229        /**
230         * {@inheritDoc}
231         */
232        @Override
233        public void setStartDate(String startDate) {
234            getElement().setAttribute("data-date-startdate", startDate);
235        }
236    
237        /**
238         * {@inheritDoc}
239         */
240        @Override
241        public void setStartDate_(Date startDate) {
242            setStartDate(dtf.format(startDate));
243        }
244    
245    
246        /**
247         * {@inheritDoc}
248         */
249        @Override
250        public void setEndDate(String endDate) {
251            getElement().setAttribute("data-date-enddate", endDate);
252        }
253    
254        /**
255         * {@inheritDoc}
256         */
257        @Override
258        public void setEndDate_(Date endDate) {
259            setEndDate(dtf.format(endDate));
260        }
261    
262        /**
263         * {@inheritDoc}
264         */
265        @Override
266        public void setAutoClose(boolean autoclose) {
267            getElement().setAttribute("data-date-autoclose", autoclose + "");
268        }
269    
270        /**
271         * {@inheritDoc}
272         */
273        @Override
274        public void setStartView(ViewMode mode) {
275            setStartView(mode.name());
276        }
277    
278        /**
279         * {@inheritDoc}
280         */
281        @Override
282        public void setStartView(String mode) {
283            getElement().setAttribute("data-date-startview", mode.toLowerCase());
284        }
