016    package com.github.gwtbootstrap.datepicker.client.ui.resources;
017    
018    import com.github.gwtbootstrap.client.ui.resources.JavaScriptInjector;
019    import com.github.gwtbootstrap.datepicker.client.ui.util.LocaleUtil;
020    import com.google.gwt.resources.client.TextResource;
021    
022    /**
023     * Utility class to inject our resources into modules page. Use it to inject
024     * JavaScript and CSS files.
025     *
026     * @author Carlos Alexandro Becker
027     * @since 2.0.3.0
028     */
029    public class DatepickerResourceInjector {
030    
031    
032        /**
033         * Injects the required CSS styles and JavaScript files into the document
034         * header.
035         */
036        public static void configure() {
037            Resources r = Resources.RESOURCES;
038            injectJs(r.datepickerJs());
039    
040            TextResource locale = LocaleUtil.getLocaleJsResource();
041            if (locale != null) {
042                JavaScriptInjector.inject(locale.getText());
043            }
044        }
045    
046        private static void injectJs(TextResource r) {
047            JavaScriptInjector.inject(r.getText());
048        }
049    }
