001package com.github.gwtbootstrap.client.ui.constants;
002
003import com.github.gwtbootstrap.client.ui.ButtonGroup;
004import com.github.gwtbootstrap.client.ui.base.Style;
005
006/**
007 * Type of a {@link ButtonGroup} toggle behavior.
008 * 
009 * @author Carlos A Becker
010 * @since 2.0.3.0
011 * 
012 */
013public enum ToggleType implements Style {
014        NONE(""), RADIO("radio"), CHECKBOX("checkbox");
015
016        private String className;
017
018        private ToggleType(String className) {
019                this.className = className;
020        }
021
022        public String get() {
023                if (className == null || className.trim().isEmpty())
024                        return "";
025                else
026                        return "buttons-" + className;
027        }
028
029}
