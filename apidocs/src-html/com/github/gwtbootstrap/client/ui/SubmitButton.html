001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.HasType;
004import com.github.gwtbootstrap.client.ui.base.StyleHelper;
005import com.github.gwtbootstrap.client.ui.constants.ButtonType;
006import com.github.gwtbootstrap.client.ui.constants.Constants;
007import com.github.gwtbootstrap.client.ui.resources.ButtonSize;
008import com.google.gwt.dom.client.Document;
009import com.google.gwt.dom.client.Element;
010import com.google.gwt.dom.client.InputElement;
011import com.google.gwt.event.dom.client.ClickHandler;
012import com.google.gwt.safehtml.shared.SafeHtml;
013import com.google.gwt.user.client.ui.ButtonBase;
014
015public class SubmitButton extends ButtonBase implements HasType<ButtonType>{
016
017        public SubmitButton() {
018                super(Document.get().createSubmitInputElement());
019                setStyleName(Constants.BTN);
020        }
021
022        protected SubmitButton(Element element) {
023                super(element);
024                assert "submit".equalsIgnoreCase(element.<InputElement> cast().getType());
025        }
026
027        public SubmitButton(SafeHtml html,
028                ClickHandler handler) {
029                this(html.asString(), handler);
030        }
031
032        public SubmitButton(SafeHtml html) {
033                this(html.asString());
034        }
035
036        public SubmitButton(String html,
037                ClickHandler handler) {
038                this(html);
039                addClickHandler(handler);
040        }
041
042        /**
043         * Creates a button with the given HTML caption.
044         * 
045         * @param html
046         *            the HTML caption
047         */
048        public SubmitButton(String html) {
049                this();
050                setHTML(html);
051        }
052
053        /**
054         * Programmatic equivalent of the user clicking the button.
055         */
056        public void click() {
057                getElement().<InputElement>cast().click();
058        }
059        
060        /**
061         * Sets the type of the Button.
062         * <p>
063         * Different types give the button a different look.
064         * 
065         * @param type
066         *            the type of the Button.
067         */
068        public void setType(ButtonType type) {
069                StyleHelper.changeStyle(this, type, ButtonType.class);
070        }
071
072        /**
073         * Sets the size of the Button.
074         * 
075         * @param size
076         *            the size of the Button.
077         */
078        public void setSize(ButtonSize size) {
079                StyleHelper.changeStyle(this, size, ButtonSize.class);
080        }
081}
