001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.IconAnchor;
004import com.github.gwtbootstrap.client.ui.constants.Constants;
005import com.google.gwt.user.client.DOM;
006import com.google.gwt.user.client.Element;
007
008/**
009 * The TabLink for {@link TabPanel}
010 * @author ohashi keisuke
011 *
012 */
013public class TabLink extends NavWidget {
014
015    private TabPane pane;
016    private boolean createTabPane = true;
017
018    /**
019     * Create widget with set Effective TabPane 
020     * @param pane effective tabPane
021     */
022    public TabLink(TabPane pane) {
023        this();
024        setText(pane.getHeading());
025        setTablePane(pane);
026    }
027
028    /**
029     * Create empty widget
030     */
031    public TabLink() {
032        super();
033        IconAnchor anchor = getAnchor();
034        anchor.getElement().setAttribute(Constants.DATA_TOGGLE, "tab");
035    }
036    
037    public void setCreateTabPane(boolean createTabPane) {
038        this.createTabPane = createTabPane;
039    }
040    
041    public boolean isCreateTabPane() {
042        return this.createTabPane;
043    }
044
045    /**
046     * Set Effective TabPane
047     * @param pane
048     */
049    public void setTablePane(TabPane pane) {
050        this.pane = pane;
051        
052        if(pane.getId() == null || pane.getId().isEmpty()) {
053            pane.setHref(DOM.createUniqueId());
054        }
055
056        setDataTarget(pane.getId());
057        
058        this.setActive(pane.isActive());
059    }
060    
061    public void setDataTarget(String id) {
062        getAnchor().getElement().setAttribute(Constants.DATA_TARGET,"#" + id);
063    }
064    
065    /**
066     * Get Effective TabPane
067     * @return effective TabPane
068     */
069    public TabPane getTabPane() {
070        return pane;
071    }
072    
073    @Override
074    protected void onAttach() {
075        super.onAttach();
076        
077        if(isActive()) {
078            show();
079        }
080    }
081    
082    @Override
083    public void setActive(boolean active) {
084        super.setActive(active);
085        
086        if(pane != null) {
087            pane.setActive(active);
088        }
089        
090    }
091    
092    /**
093     * show tab pane
094     */
095    public void show() {
096        setActive(true);
097        show(getAnchor().getElement());
098    }
099
100    //@formatter:off
101    private native void show(Element e)/*-{
102        $wnd.jQuery(e).tab('show');
103    }-*/;
104    //@formatter:on
105
106}
