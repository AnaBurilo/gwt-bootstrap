001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.TextBoxBase;
004import com.google.gwt.dom.client.Document;
005import com.google.gwt.dom.client.Element;
006import com.google.gwt.dom.client.InputElement;
007import com.google.gwt.user.client.ui.RootPanel;
008import com.google.gwt.user.client.ui.Widget;
009
010/**
011 * A TextBox for Bootstrap form.
012 * 
013 * @since 2.0.3.0
014 * 
015 * @author Carlos Alexandro Becker
016 * @author ohashi keisuke
017 * 
018 */
019public class TextBox extends TextBoxBase {
020
021        /**
022         * Creates a TextBox widget that wraps an existing &lt;input type='text'&gt;
023         * element.
024         * 
025         * This element must already be attached to the document. If the element is
026         * removed from the document, you must call
027         * {@link RootPanel#detachNow(Widget)}.
028         * 
029         * @param element
030         *            the element to be wrapped
031         */
032        public static TextBox wrap(Element element) {
033                // Assert that the element is attached.
034                assert Document.get().getBody().isOrHasChild(element);
035
036                TextBox textBox = new TextBox(element);
037
038                // Mark it attached and remember it for cleanup.
039                textBox.onAttach();
040                RootPanel.detachOnWindowClose(textBox);
041
042                return textBox;
043        }
044
045        /**
046         * Creates an empty text box.
047         */
048        public TextBox() {
049                this(Document.get().createTextInputElement(), "gwt-TextBox");
050        }
051
052        /**
053         * This constructor may be used by subclasses to explicitly use an existing
054         * element. This element must be an &lt;input&gt; element whose type is
055         * 'text'.
056         * 
057         * @param element
058         *            the element to be used
059         */
060        protected TextBox(Element element) {
061                super(element);
062                assert InputElement.as(element).getType().equalsIgnoreCase("text");
063        }
064
065        TextBox(Element element,
066                String styleName) {
067                super(element);
068                if (styleName != null) {
069                        setStyleName(styleName);
070                }
071        }
072
073        /**
074         * Gets the maximum allowable length of the text box.
075         * 
076         * @return the maximum length, in characters
077         */
078        public int getMaxLength() {
079                return getInputElement().getMaxLength();
080        }
081
082        /**
083         * Gets the number of visible characters in the text box.
084         * 
085         * @return the number of visible characters
086         */
087        public int getVisibleLength() {
088                return getInputElement().getSize();
089        }
090
091        /**
092         * Sets the maximum allowable length of the text box.
093         * 
094         * @param length
095         *            the maximum length, in characters
096         */
097        public void setMaxLength(int length) {
098                getInputElement().setMaxLength(length);
099        }
100
101        /**
102         * Sets the number of visible characters in the text box.
103         * 
104         * @param length
105         *            the number of visible characters
106         */
107        public void setVisibleLength(int length) {
108                getInputElement().setSize(length);
109        }
110
111        private InputElement getInputElement() {
112                return getElement().cast();
113        }
114
115}
