016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.resources.Bootstrap;
019import com.github.gwtbootstrap.client.ui.resources.prettify.HasProgrammingLanguage;
020import com.github.gwtbootstrap.client.ui.resources.prettify.PrettifyHelper;
021import com.google.gwt.safehtml.shared.SafeHtmlBuilder;
022import com.google.gwt.user.client.ui.HTMLPanel;
023
024/**
025 * A block of Code
026 * 
027 * @author Dominik Mayer
028 */
029public class CodeBlock extends HTMLPanel implements HasProgrammingLanguage {
030
031        private final PrettifyHelper helper;
032        private boolean linenums = false;
033        
034        public CodeBlock() {
035                super("pre", "");
036                helper = new PrettifyHelper(this);
037        }
038
039        public CodeBlock(String html) {
040                super("pre", html);
041                helper = new PrettifyHelper(this);
042                setHTML(html);
043        }
044
045        public void setScrollable(boolean scrollable) {
046                if (scrollable)
047                        addStyleName(Bootstrap.pre_scrollable);
048                else
049                        removeStyleName(Bootstrap.pre_scrollable);
050        }
051
052        public void setLinenums(boolean linenums) {
053                this.linenums = linenums;
054        }
055
056        public String getHTML() {
057                return getElement().getInnerHTML();
058        }
059
060        public void setHTML(String html) {
061                String[] lines = html.split("\\\\n");
062                SafeHtmlBuilder shb = new SafeHtmlBuilder();
063
064                for (String s : lines) {
065                        shb.appendEscaped(s);
066                        shb.appendHtmlConstant("<br/>");
067                }
068
069                getElement().setInnerHTML(shb.toSafeHtml().asString());
070        }
071
072        public String getText() {
073                return getHTML();
074        }
075
076        public void setText(String text) {
077                setHTML(text);
078        }
079
080        public void setLang(String lang) {
081                helper.setLang(lang);
082        }
083        
084        @Override
085        protected void onLoad() {
086                super.onLoad();
087                helper.configure(linenums);
088        }
089}
