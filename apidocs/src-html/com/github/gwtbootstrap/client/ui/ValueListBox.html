018import java.util.ArrayList;
019import java.util.Collection;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023
024import com.github.gwtbootstrap.client.ui.base.HasAlternateSize;
025import com.github.gwtbootstrap.client.ui.base.HasId;
026import com.github.gwtbootstrap.client.ui.base.HasSize;
027import com.github.gwtbootstrap.client.ui.base.IsSearchQuery;
028import com.github.gwtbootstrap.client.ui.base.SearchQueryStyleHelper;
029import com.github.gwtbootstrap.client.ui.base.SizeHelper;
030import com.github.gwtbootstrap.client.ui.base.StyleHelper;
031import com.github.gwtbootstrap.client.ui.constants.AlternateSize;
032import com.github.gwtbootstrap.client.ui.constants.Constants;
033import com.google.gwt.editor.client.IsEditor;
034import com.google.gwt.editor.client.adapters.TakesValueEditor;
035import com.google.gwt.event.dom.client.ChangeEvent;
036import com.google.gwt.event.dom.client.ChangeHandler;
037import com.google.gwt.event.logical.shared.ValueChangeEvent;

039import com.google.gwt.event.shared.HandlerRegistration;

048/**
049 * The TW Bootstrap style ValueListBox
050 * 
051 * @param <T> the value type
052 * @author ohashi keisuke
053 * @since 2.0.4.0
054 */
055public class ValueListBox<T> extends Composite implements HasConstrainedValue<T>, IsEditor<TakesValueEditor<T>>, HasName, HasId, HasEnabled, HasSize, HasAlternateSize, IsSearchQuery {
056
057        private final Map<Object, Integer> valueKeyToIndex = new HashMap<Object, Integer>();
058
059        private final List<T> valueList = new ArrayList<T>();
060
061        private final Renderer<T> renderer;
062
063        private final ProvidesKey<T> keyProvider;
064
065        private TakesValueEditor<T> editor;
066
067        private T value;
068
069        public ValueListBox(Renderer<T> renderer) {
070                this(renderer, new SimpleKeyProvider<T>());
071        }
072
073        public ValueListBox(Renderer<T> renderer,
074                ProvidesKey<T> keyProvider) {
075                this.keyProvider = keyProvider;
076                this.renderer = renderer;
077                initWidget(new ListBox());
078
079                getListBox().addChangeHandler(new ChangeHandler() {
080
081                        public void onChange(ChangeEvent event) {
082                                int selectedIndex = getListBox().getSelectedIndex();
083
084                                if (selectedIndex < 0) {
085                                        return;
086                                }
087                                T newValue = valueList.get(selectedIndex);
088                                setValue(newValue, true);
089                        }
090                });
091        }
092
093        public void setName(String name) {
094                getListBox().setName(name);
095        }
096
097        public String getName() {
098                return getListBox().getName();
099        }
100
101        public HandlerRegistration addValueChangeHandler(ValueChangeHandler<T> handler) {
102                return addHandler(handler, ValueChangeEvent.getType());
103        }
104
105        /**
106         * Returns a {@link TakesValueEditor} backed by the ValueListBox.
107         */
108        public TakesValueEditor<T> asEditor() {
109                if (editor == null) {
110                        editor = TakesValueEditor.of(this);
111                }
112                return editor;
113        }
114
115        public T getValue() {
116                return value;
117        }
118
119        public void setAcceptableValues(Collection<T> newValues) {
120
121                valueList.clear();
122                
123                valueKeyToIndex.clear();
124                
125                ListBox listBox = getListBox();
126                
127                listBox.clear();
128
129                for (T nextNewValue : newValues) {
130                        addValue(nextNewValue);
131                }
132
133                updateListBox();
134        }
135
136        /**
137         * Set the value and display it in the select element. Add the value to the
138         * acceptable set if it is not already there.
139         */
140        public void setValue(T value) {
141                setValue(value, false);
142        }
143
144        public void setValue(T value, boolean fireEvents) {
145                if (value == this.value || (this.value != null && this.value.equals(value))) {
146                        return;
147                }
148
149                T before = this.value;
150                this.value = value;
151                updateListBox();
152
153                if (fireEvents) {
154                        ValueChangeEvent.fireIfNotEqual(this, before, value);
155                }
156        }
157
158        private void addValue(T value) {
159                Object key = keyProvider.getKey(value);
160                if (valueKeyToIndex.containsKey(key)) {
161                        throw new IllegalArgumentException("Duplicate value: " + value);
162                }
163
164                valueKeyToIndex.put(key, valueList.size());
165                valueList.add(value);
166                getListBox().addItem(renderer.render(value));
167                assert valueList.size() == getListBox().getItemCount();
168        }
169
170        private ListBox getListBox() {
171                return (ListBox) getWidget();
172        }
173
174        private void updateListBox() {
175                Object key = keyProvider.getKey(value);
176                Integer index = valueKeyToIndex.get(key);
177                if (index == null) {
178                        addValue(value);
179                }
180
181                index = valueKeyToIndex.get(key);
182                getListBox().setSelectedIndex(index);
183        }
184
185        /**
186         * {@inheritDoc}
187         */
188        @Override
189        public String getId() {
190                return getListBox().getId();
191        }
192
193        /**
194         * {@inheritDoc}
195         */
196        @Override
197        public void setId(String id) {
198                getListBox().setId(id);
199        }
200
201        /**
202         * {@inheritDoc}
203         */
204        @Override
205        public void setSearchQuery(boolean searchQuery) {
206                SearchQueryStyleHelper.setSearchQuery(this, searchQuery);
207        }
208
209        /**
210         * {@inheritDoc}
211         */
212        @Override
213        public boolean isSearchQuery() {
214                return SearchQueryStyleHelper.isSearchQuery(this);
215        }
216
217
218        /**
219         * {@inheritDoc}
220         */
221        @Override
222        public void setAlternateSize(AlternateSize size) {
223                StyleHelper.changeStyle(this, size, AlternateSize.class);
224        }
225
226        /**
227         * {@inheritDoc}
228         */
229        @Override
230        public void setSize(int size) {
231                SizeHelper.setSize(this, size);
232        }
233        /**
234         * {@inheritDoc}
235         */
236        @Override
237        public void setEnabled(boolean enabled) {
238                getListBox().setEnabled(enabled);
239                if(enabled) {
240                        removeStyleName(Constants.DISABLED);
241                } else {
242                        addStyleName(Constants.DISABLED);
243                }
244        }
245
246        @Override
247        public boolean isEnabled() {
248                return getListBox().isEnabled();
249        }
250}
