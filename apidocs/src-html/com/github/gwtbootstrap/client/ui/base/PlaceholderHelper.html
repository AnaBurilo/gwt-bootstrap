001    package com.github.gwtbootstrap.client.ui.base;
002    
003    import com.github.gwtbootstrap.client.ui.constants.Constants;
004    import com.google.gwt.user.client.Element;
005    
006    
007    /**
008     * The helper class for {@link HasPlaceholder} interface.
009     * 
010     * @author ohashi keisuke
011     *
012     */
013    public class PlaceholderHelper {
014    
015            //TODO 2012/05/04 ohashi keisuke. Support for non HTML5 Browser.
016            //I designed this class as non static class.
017            //Maybe we can use deffered binding for non HTML5 Browser support.
018            
019            /**
020             * Set a placeholder attribute (HTML5) to the element.
021             * 
022             * @param element the element be set 
023             * @param placeholder
024             *            the String to show. Use a empty string or null to remove the attribute.
025             */
026            public void setPlaceholer(Element element , String placeholder) {
027                    if (placeholder == null || placeholder.trim().isEmpty()) {
028                            element.removeAttribute(Constants.PLACEHOLDER);
029                            return;
030                    }
031                    element.setAttribute(Constants.PLACEHOLDER, placeholder);
032            }
033            
034            /**
035             * Get a placeholder attribute (HTML5) from the element.
036             * <p>
037             * If the element don't have placeholder attribute,This return empty string.
038             * </p>
039             * @param element the element
040             * @return placeholder attribute.
041             */
042            public String getPlaceholder(Element element) {
043                    return element.hasAttribute(Constants.PLACEHOLDER) 
044                                    ? element.getAttribute(Constants.PLACEHOLDER)
045                                    : "";
046            }
047    }
