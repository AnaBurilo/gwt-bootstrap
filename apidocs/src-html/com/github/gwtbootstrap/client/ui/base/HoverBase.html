016package com.github.gwtbootstrap.client.ui.base;
017
018import com.github.gwtbootstrap.client.ui.constants.Placement;
019import com.github.gwtbootstrap.client.ui.constants.Trigger;
020import com.github.gwtbootstrap.client.ui.constants.VisibilityChange;
021import com.google.gwt.user.client.ui.HasText;
022
023//@formatter:off
024/**
025* Base class for widgets that hover above other widgets.
026* 
027* @since 2.0.3.0
028* 
029* @author Dominik Mayer
030* 
031* @see <a href="http://twitter.github.com/bootstrap/javascript.html#popovers">Bootstrap documentation</a>
032*/
033//@formatter:on
034public abstract class HoverBase extends ComplexWidget implements IsAnimated,
035                HasTrigger, HasPlacement, HasText, HasShowDelay, HasVisibility {
036
037        /**
038         * Whether the widget is animated or not.
039         */
040        protected boolean animated = true;
041
042        /**
043         * The placement of the widget relative to its trigger element.
044         */
045        protected Placement placement = Placement.TOP;
046
047        /**
048         * The action that triggers the widget.
049         */
050        protected Trigger trigger = Trigger.HOVER;
051
052        /**
053         * The delay until the widget is shown.
054         */
055        protected int showDelayInMilliseconds = 0;
056
057        /**
058         * The delay until the widget is hidden.
059         */
060        protected int hideDelayInMilliseconds = 0;
061
062        /**
063         * Creates a new widget based on the provided HTML tag.
064         */
065        public HoverBase() {
066                super("span");
067        }
068
069        /**
070         * {@inheritDoc}
071         */
072        @Override
073        protected void onLoad() {
074                super.onLoad();
075                reconfigure();
076        }
077
078        /**
079         * Adds an HTML data attribute to the widget's tag.
080         * 
081         * @param attribute
082         *            the name of the attribute without leading <code>"data-"</code>
083         * @param value
084         *            the value to be stored
085         */
086        protected void setDataAttribute(String attribute, String value) {
087                getElement().setAttribute("data-" + attribute, value);
088        }
089
090        /**
091         * Returns the data stored in one of the widget's HTML data attributes.
092         * 
093         * @param attribute
094         *            the name of the attribute without leading <code>"data-"</code>
095         * @return the value stored in the tag
096         */
097        protected String getDataAttribute(String attribute) {
098                return getElement().getAttribute("data-" + attribute);
099        }
100
101        /**
102         * {@inheritDoc}
103         */
104        public void setAnimation(boolean animated) {
105                this.animated = animated;
106        }
107
108        /**
109         * Redraws the widget with the currently set options. This must <b>not</b>
110         * be called when a parameter is updated because it would deactivate all
111         * other parameters. No idea why...
112         */
113        protected abstract void reconfigure();
114
115        /**
116         * {@inheritDoc}
117         */
118        public boolean getAnimation() {
119                return animated;
120        }
121
122        /**
123         * {@inheritDoc} Relative to its trigger element.
124         */
125        public void setPlacement(Placement placement) {
126                this.placement = placement;
127        }
128
129        /**
130         * {@inheritDoc}
131         */
132        public Placement getPlacement() {
133                return placement;
134        }
135
136        /**
137         * {@inheritDoc}
138         */
139        public void setTrigger(Trigger trigger) {
140                this.trigger = trigger;
141        }
142
143        /**
144         * {@inheritDoc}
145         */
146        public Trigger getTrigger() {
147                return trigger;
148        }
149
150        /**
151         * {@inheritDoc}
152         */
153        public void setShowDelay(int delayInMilliseconds) {
154                showDelayInMilliseconds = delayInMilliseconds;
155        }
156
157        /**
158         * {@inheritDoc}
159         */
160        public int getShowDelay() {
161                return showDelayInMilliseconds;
162        }
163
164        /**
165         * {@inheritDoc}
166         */
167        public void setHideDelay(int delayInMilliseconds) {
168                hideDelayInMilliseconds = delayInMilliseconds;
169        }
170
171        /**
172         * {@inheritDoc}
173         */
174        public int getHideDelay() {
175                return hideDelayInMilliseconds;
176        }
177
178        /**
179         * {@inheritDoc}
180         */
181        public void show() {
182                changeVisibility(VisibilityChange.SHOW);
183        }
184
185        /**
186         * {@inheritDoc}
187         */
188        public void hide() {
189                changeVisibility(VisibilityChange.HIDE);
190        }
191
192        /**
193         * {@inheritDoc}
194         */
195        public void toggle() {
196                changeVisibility(VisibilityChange.TOGGLE);
197        }
198
199        /**
200         * Changes the visibility of the widget.
201         * 
202         * @param visibilityChange
203         *            the action to be performed
204         */
205        protected abstract void changeVisibility(VisibilityChange visibilityChange);
206
207}
