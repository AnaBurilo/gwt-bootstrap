016package com.github.gwtbootstrap.client.ui.base;
017
018import com.github.gwtbootstrap.client.ui.Close;
019import com.github.gwtbootstrap.client.ui.constants.AlertType;
020import com.github.gwtbootstrap.client.ui.constants.Constants;
021import com.github.gwtbootstrap.client.ui.constants.DismissType;
022import com.github.gwtbootstrap.client.ui.event.CloseEvent;
023import com.github.gwtbootstrap.client.ui.event.CloseHandler;
024import com.github.gwtbootstrap.client.ui.event.ClosedEvent;
025import com.github.gwtbootstrap.client.ui.event.ClosedHandler;
026import com.github.gwtbootstrap.client.ui.event.HasCloseHandlers;
027import com.google.gwt.dom.client.Element;
028import com.google.gwt.event.shared.HandlerRegistration;
029import com.google.gwt.safehtml.shared.SafeHtmlUtils;
030import com.google.gwt.user.client.ui.HasHTML;
031
032/**
033 * Base class for Alert widgets.
034 * 
035 * @since 2.0.3.0
036 * 
037 * @author Dominik Mayer
038 * @author Carlos Alexandro Becker
039 * 
040 * @see <a
041 *      href="http://twitter.github.com/bootstrap/components.html#alerts">Bootstrap
042 *      documentation</a>
043 */
044public abstract class AlertBase extends DivWidget implements HasHTML,
045                IsAnimated, HasCloseHandlers, HasType<AlertType> {
046
047        private Close close;
048
049        private String html = "";
050
051        private String heading = "";
052
053        private boolean fade;
054
055        /**
056         * Initializes an Alert with a close icon.
057         */
058        public AlertBase() {
059                this(true);
060        }
061
062        /**
063         * Initializes an Alert with an optional close icon.
064         * 
065         * @param hasClose
066         *            whether the Alert should have a close icon.
067         */
068        public AlertBase(boolean hasClose) {
069                super.setStyleName(Constants.ALERT);
070                setClose(hasClose);
071                setHandlerFunctions(getElement());
072        }
073
074        /**
075         * Sets whether the Alert has a close icon or not.
076         * 
077         * @param hasClose
078         *            <code>false</code> if you don't want to have a close icon.
079         *            Default: <code>true</code>
080         */
081        public void setClose(boolean hasClose) {
082                if (hasClose) {
083                        close = new Close(DismissType.ALERT);
084                        add(close);
085                } else {
086                        clear();
087                        close = null;
088                }
089                redraw();
090        }
091
092        /**
093         * Adds the Java functions that fire the Events to document. It is a
094         * convenience method to have a cleaner code later on.
095         */
096        private native void setHandlerFunctions(Element e) /*-{
097                var that = this;
098                $wnd
099                                .jQuery(e)
100                                .bind(
101                                                'close',
102                                                function() {
103                                                        that.@com.github.gwtbootstrap.client.ui.base.AlertBase::onClose()();
104                                                });
105                $wnd
106                                .jQuery(e)
107                                .bind(
108                                                'closed',
109                                                function() {
110                                                        that.@com.github.gwtbootstrap.client.ui.base.AlertBase::onClosed()();
111                                                });
112        }-*/;
113
114        /**
115         * This method is called immediately when the widget's close method is
116         * executed.
117         */
118        protected void onClose() {
119                fireEvent(new CloseEvent());
120        }
121
122        /**
123         * This method is called once the widget is completely closed.
124         */
125        protected void onClosed() {
126                fireEvent(new ClosedEvent());
127        }
128
129        /**
130         * Initializes an Alert of given Type with a close icon.
131         * 
132         * @param type
133         *            of the Alert
134         */
135        public AlertBase(AlertType type) {
136                this();
137                setType(type);
138        }
139
140        /**
141         * Sets the type of the Alert.
142         * 
143         * @param type
144         */
145        public void setType(AlertType type) {
146                super.setStyleName(Constants.ALERT);
147                super.addStyle(type);
148                setFade();
149        }
150
151        /**
152         * Sets the text of an optional heading. The implementation depends on the
153         * subclass.
154         * 
155         * @param text
156         *            the new heading
157         */
158        public void setHeading(String text) {
159                heading = text;
160                redraw();
161        }
162
163        /**
164         * Sets whether the Alert should be animated.
165         * 
166         * @param animated
167         *            <code>true</code> if the Alert should fade out. Default:
168         *            <code>false</code>
169         */
170        public void setAnimation(boolean animated) {
171                this.fade = animated;
172                setFade();
173        }
174
175        public boolean getAnimation() {
176                return fade;
177        }
178
179        /**
180         * Delete the whole content of the Alert. This includes text, heading and
181         * close icon.
182         */
183        @Override
184        public void clear() {
185                getElement().setInnerHTML("");
186        }
187
188        /**
189         * Redraws the Alert according to the specified settings.
190         */
191        protected void redraw() {
192                setHTML(html);
193                configure(getElement());
194        }
195
196        /**
197         * Sets the classes that define whether the Alert fades or not.
198         */
199        private void setFade() {
200                if (fade) {
201                        addStyleName("fade");
202                        addStyleName("in");
203                } else {
204                        removeStyleName("fade");
205                        removeStyleName("in");
206                }
207        }
208
209        /**
210         * {@inheritDoc}
211         */
212        public String getText() {
213                return getHTML(); // TODO
214        }
215
216        /**
217         * {@inheritDoc}
218         */
219        public void setText(String text) {
220                setHTML(SafeHtmlUtils.htmlEscape(text));
221        }
222
223        /**
224         * {@inheritDoc}
225         */
226        public String getHTML() {
227                return html;
228        }
229
230        /**
231         * {@inheritDoc}
232         */
233        public void setHTML(String html) {
234                String output = "";
235                if (close != null)
236                        output = close.toString();
237                output = output + heading + html;
238                setHTMLOutput(html, output);
239        }
240
241        /**
242         * Sets the HTML of the Alert. Override this method to implement a custom
243         * formatting.
244         * 
245         * @param html
246         *            the html of the content without heading and close icon.
247         * @param output
248         *            the raw output to be put into the widget. Be careful!
249         */
250        protected void setHTMLOutput(String html, String output) {
251                this.html = html;
252                getElement().setInnerHTML(output);
253        }
254
255        /**
256         * {@inheritDoc}
257         */
258        @Override
259        protected void onLoad() {
260                super.onLoad();
261                configure(getElement());
262        }
263
264        private native void configure(Element e) /*-{
265                $wnd.jQuery(e).alert(e);
266        }-*/;
267
268        /**
269         * {@inheritDoc}
270         */
271        public HandlerRegistration addCloseHandler(CloseHandler handler) {
272                return addHandler(handler, CloseEvent.getType());
273        }
274
275        /**
276         * {@inheritDoc}
277         */
278        public HandlerRegistration addClosedHandler(ClosedHandler handler) {
279                return addHandler(handler, ClosedEvent.getType());
280        }
281}
