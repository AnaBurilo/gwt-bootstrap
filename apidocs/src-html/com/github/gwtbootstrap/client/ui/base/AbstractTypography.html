016    package com.github.gwtbootstrap.client.ui.base;
017    
018    import com.github.gwtbootstrap.client.ui.constants.Device;
019    import com.github.gwtbootstrap.client.ui.constants.ResponsiveStyle;
020    import com.google.gwt.user.client.ui.HasText;
021    import com.google.gwt.user.client.ui.Widget;
022    
023    /**
024     * Helper class for widgets that have text.
025     * 
026     * @since 2.0.2.0
027     * 
028     * @author Carlos Alexandro Becker
029     * 
030     */
031    public abstract class AbstractTypography extends Widget implements HasText,
032                    HasStyle, IsResponsive {
033    
034            /**
035             * {@inheritDoc}
036             */
037            public void setText(String text) {
038                    getElement().setInnerText(text);
039            }
040    
041            /**
042             * {@inheritDoc}
043             */
044            public String getText() {
045                    return getElement().getInnerText();
046            }
047    
048            /**
049             * {@inheritDoc}
050             */
051            public void setStyle(Style style) {
052                    setStyleName(style.get());
053            }
054    
055            /**
056             * {@inheritDoc}
057             */
058            public void addStyle(Style style) {
059                    addStyleName(style.get());
060            }
061    
062            /**
063             * {@inheritDoc}
064             */
065            public void removeStyle(Style style) {
066                    String styleString = style.get();
067                    if (!styleString.isEmpty())
068                            removeStyleName(styleString);
069            }
070    
071            /**
072             * {@inheritDoc}
073             */
074            public void setShowOn(Device device) {
075                    removeResponsiveStyles();
076                    switch (device) {
077                            case PHONE:
078                                    addStyle(ResponsiveStyle.VISIBLE_PHONE);
079                                    break;
080                            case TABLET:
081                                    addStyle(ResponsiveStyle.VISIBLE_TABLET);
082                                    break;
083                            case DESKTOP:
084                                    addStyle(ResponsiveStyle.VISIBLE_DESKTOP);
085                                    break;
086                    }
087            }
088    
089            private void removeResponsiveStyles() {
090                    removeStyle(ResponsiveStyle.VISIBLE_PHONE);
091                    removeStyle(ResponsiveStyle.VISIBLE_TABLET);
092                    removeStyle(ResponsiveStyle.VISIBLE_DESKTOP);
093                    removeStyle(ResponsiveStyle.HIDDEN_PHONE);
094                    removeStyle(ResponsiveStyle.HIDDEN_TABLET);
095                    removeStyle(ResponsiveStyle.HIDDEN_DESKTOP);
096            }
097    
098            /**
099             * {@inheritDoc}
100             * 
101             * @deprecated This method should never be called directly. It will break
102             *             your implementation if any style names change. The only valid
103             *             use is inside UiBinder files where it processes the
104             *             <code>showOn="..."</code> argument. Use
105             *             {@link #setShowOn(Device)} instead!
106             */
107            @Deprecated
108            public void setShowOn(String device) {
109                    if (device.equalsIgnoreCase("phone")) {
110                            setShowOn(Device.PHONE);
111                    } else if (device.equalsIgnoreCase("tablet")) {
112                            setShowOn(Device.TABLET);
113                    } else if (device.equalsIgnoreCase("desktop")) {
114                            setShowOn(Device.DESKTOP);
115                    }
116            }
117    
118            /**
119             * {@inheritDoc}
120             */
121            public void setHideOn(Device device) {
122                    removeResponsiveStyles();
123                    switch (device) {
124                            case PHONE:
125                                    addStyle(ResponsiveStyle.HIDDEN_PHONE);
126                                    break;
127                            case TABLET:
128                                    addStyle(ResponsiveStyle.HIDDEN_TABLET);
129                                    break;
130                            case DESKTOP:
131                                    addStyle(ResponsiveStyle.HIDDEN_DESKTOP);
132                                    break;
133                    }
134            }
135    
136            /**
137             * {@inheritDoc}
138             * 
139             * @deprecated This method should never be called directly. It will break
140             *             your implementation if any style names change. The only valid
141             *             use is inside UiBinder files where it processes the
142             *             <code>hideOn="..."</code> argument. Use
143             *             {@link #setHideOn(Device)} instead!
144             * 
145             */
146            @Deprecated
147            public void setHideOn(String device) {
148                    if (device.equalsIgnoreCase("phone")) {
149                            setHideOn(Device.PHONE);
150                    } else if (device.equalsIgnoreCase("tablet")) {
151                            setHideOn(Device.TABLET);
152                    } else if (device.equalsIgnoreCase("desktop")) {
153                            setHideOn(Device.DESKTOP);
154                    }
155            }
156    }
