016package com.github.gwtbootstrap.client.ui.base;
017
018import com.github.gwtbootstrap.client.ui.NavLink;
019import com.github.gwtbootstrap.client.ui.constants.Constants;
020import com.google.gwt.core.client.GWT;
021import com.google.gwt.dom.client.Document;
022import com.google.gwt.dom.client.Element;
023import com.google.gwt.event.dom.client.ClickEvent;
024import com.google.gwt.event.dom.client.ClickHandler;
025import com.google.gwt.event.dom.client.DomEvent;
026import com.google.gwt.event.dom.client.HasClickHandlers;
027import com.google.gwt.event.shared.HandlerRegistration;
028import com.google.gwt.user.client.ui.HasText;
029import com.google.gwt.user.client.ui.HasWidgets;
030import com.google.gwt.user.client.ui.Widget;
031
032//@formatter:off
033
034/**
035 * Base class for dropdown widgets.
036 *
037 * @author Carlos A Becker
038 * @author Dominik Mayer
039 * @since 2.0.4.0
040 */
041//@formatter:on
042public abstract class DropdownBase extends ComplexWidget implements HasClickHandlers, HasWidgets, HasText {
043
044        private UnorderedList menu = new UnorderedList();
045
046        private IconAnchor trigger;
047
048        private NavLink link;
049
050        private NavLinkClickHandler handler = new NavLinkClickHandler();
051
052        /**
053         * Creates a new widget of the given type.
054         * 
055         * @param type
056         *            the HTML tag to be used for the widget
057         */
058        public DropdownBase(String type) {
059                super(type);
060                createAndAddTrigger();
061                menu.setStyleName("dropdown-menu");
062                super.add(menu);
063        }
064
065        private void createAndAddTrigger() {
066                trigger = createTrigger();
067                trigger.addStyleName("dropdown-toggle");
068                trigger.getElement().setAttribute(Constants.DATA_TOGGLE, "dropdown");
069                super.add(trigger);
070        }
071
072        /**
073         * Sets the text of the dropdown trigger.
074         * 
075         * @param text
076         */
077        public void setText(String text) {
078                trigger.setText(text);
079        }
080
081        /**
082         * @return the text of the dropdown trigger.
083         */
084        public String getText() {
085                return trigger.getText();
086        }
087
088        /**
089         * Implement this method to create the trigger appropriate for your widget.
090         * It has to be an {@link IconAnchor} or a subtype.
091         * 
092         * @return the created trigger
093         */
094        protected abstract IconAnchor createTrigger();
095
096        /**
097         * Set dropup style.
098         * 
099         * @param dropup
100         *            true:Set Dropup false:Un-set Dropup
101         */
102        public void setDropup(boolean dropup) {
103                if (dropup)
104                        addStyleName(Constants.DROPUP);
105                else
106                        removeStyleName(Constants.DROPUP);
107        }
108
109        /**
110         * {@inheritDoc}
111         */
112        @Override
113        protected void onLoad() {
114                super.onLoad();
115                if (trigger != null) {
116                        configure(trigger.getElement());
117                }
118        }
119
120        /**
121         * Adds a widget to the dropdown menu.
122         * 
123         * @param widget
124         *            the widget that will be added to the menu
125         * @see #addWidget(Widget)
126         */
127        @Override
128        public void add(Widget widget) {
129                menu.add(widget);
130                if (widget instanceof NavLink) {
131                        ((NavLink) widget).addClickHandler(handler);
132                }
133        }
134
135        /**
136         * Adds a widget to the the dropdown widget, <b>not</b> to the dropdown
137         * menu.
138         * <p/>
139         * <p/>
140         * Use {@link #add(Widget)} if you want to add a widget to the dropdown
141         * menu.
142         * 
143         * @param widget
144         *            the widget to be added.
145         */
146        protected void addWidget(Widget widget) {
147                super.add(widget);
148        }
149
150        private native void configure(Element e) /*-{
151                                                                                                $wnd.jQuery(e).dropdown();
152                                                                                                }-*/;
153
154        /**
155         * Pull the dropdown menu to right
156         * 
157         * @param rightDropdown
158         *            <code>true</code> pull to right, otherwise to left. Default is
159         *            <code>false</code>
160         */
161        public void setRightDropdown(boolean rightDropdown) {
162                if (rightDropdown) {
163                        menu.addStyleName("pull-right");
164                } else {
165                        menu.removeStyleName("pull-right");
166                }
167        }
168
169        private class NavLinkClickHandler implements ClickHandler {
170
171                @Override
172                public void onClick(ClickEvent event) {
173                        try {
174                                IconAnchor iconAnchor = (IconAnchor) event.getSource();
175                                link = (NavLink) iconAnchor.getParent();
176                        } catch (Exception e) {
177                                GWT.log(e.getMessage() , e);
178                        }
179                        DomEvent.fireNativeEvent(Document.get().createChangeEvent(), DropdownBase.this);
180                }
181
182        }
183
184        /**
185         * Method to get the {@link NavLink} that has been clicked most recently.
186         * 
187         * @return Last clicked NavLink or <code>null</code> if none have been
188         *         clicked.
189         */
190        public NavLink getLastSelectedNavLink() {
191                return link;
192        }
193
194        @Override
195        public void clear() {
196                menu.clear();
197        }
198
199        @Override
200        public HandlerRegistration addClickHandler(ClickHandler handler) {
201                return trigger.addClickHandler(handler);
202        }
203}
