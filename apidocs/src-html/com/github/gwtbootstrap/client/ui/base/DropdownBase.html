016    package com.github.gwtbootstrap.client.ui.base;
017    
018    import com.github.gwtbootstrap.client.ui.Divider;
019    import com.github.gwtbootstrap.client.ui.NavHeader;
020    import com.github.gwtbootstrap.client.ui.NavLink;
021    import com.github.gwtbootstrap.client.ui.NavText;
022    import com.github.gwtbootstrap.client.ui.constants.Constants;
023    import com.google.gwt.dom.client.Document;
024    import com.google.gwt.dom.client.Element;
025    import com.google.gwt.event.dom.client.*;
026    import com.google.gwt.event.shared.HandlerRegistration;
027    import com.google.gwt.user.client.ui.Widget;

031    /**
032     * Base class for dropdown widgets.
033     *
034     * @author Carlos A Becker
035     * @author Dominik Mayer
036     * @since 2.0.3.0
037     */
038    //@formatter:on
039    public abstract class DropdownBase extends ComplexWidget {
040    
041        private UnorderedList menu = new UnorderedList();
042    
043        private IconAnchor trigger;
044    
045        private NavLink link;
046    
047        private NavLinkClickHandler handler = new NavLinkClickHandler();
048    
049        /**
050         * Creates a new widget of the given type.
051         *
052         * @param type the HTML tag to be used for the widget
053         */
054        public DropdownBase(String type) {
055            super(type);
056            createAndAddTrigger();
057            menu.setStyleName("dropdown-menu");
058            super.add(menu);
059        }
060    
061        private void createAndAddTrigger() {
062            trigger = createTrigger();
063            trigger.addStyleName("dropdown-toggle");
064            trigger.getElement().setAttribute(Constants.DATA_TOGGLE, "dropdown");
065            super.add(trigger);
066        }
067    
068        /**
069         * Sets the text of the dropdown trigger.
070         *
071         * @param text
072         */
073        public void setText(String text) {
074            trigger.setText(text);
075        }
076    
077        /**
078         * Implement this method to create the trigger appropriate for your widget.
079         * It has to be an {@link IconAnchor} or a subtype.
080         *
081         * @return the created trigger
082         */
083        protected abstract IconAnchor createTrigger();
084    
085        public void setDropup(boolean dropup) {
086            if (dropup)
087                addStyleName(Constants.DROPUP);
088            else
089                removeStyleName(Constants.DROPUP);
090        }
091    
092        /**
093         * {@inheritDoc}
094         */
095        @Override
096        protected void onLoad() {
097            super.onLoad();
098            if (trigger != null) {
099                configure(trigger.getElement());
100            }
101        }
102    
103        /**
104         * Adds a widget to the dropdown menu.
105         *
106         * @param widget the widget that will be added to the menu
107         * @see #addWidget(Widget)
108         */
109        @Override
110        public void add(Widget widget) {
111            if (canBeAdded(widget)) {
112                menu.add(widget);
113                if (widget instanceof NavLink) {
114                    ((NavLink) widget).addClickHandler(handler);
115                }
116            } else {
117                throw new IllegalArgumentException("Only NavLink, NavText and "
118                        + "Divider can be added to the Dropdown");
119            }
120        }
121    
122        /**
123         * Checks whether a widget can be added to the dropdown.
124         * <p/>
125         * By default, the dropdown accepts {@link NavLink}, {@link NavText} and
126         * {@link Divider} widgets. Override this class to allow more widgets.
127         *
128         * @param widget the widget that should be added
129         * @return <code>true</code> if the widget can be added to the Dropdown
130         */
131        protected boolean canBeAdded(Widget widget) {
132            return ((widget instanceof NavLink) || (widget instanceof NavText)
133                    || (widget instanceof NavHeader) || (widget instanceof Divider));
134        }
135    
136        /**
137         * Adds a widget to the the dropdown widget, <b>not</b> to the dropdown
138         * menu.
139         * <p/>
140         * <p/>
141         * Use {@link #add(Widget)} if you want to add a widget to the dropdown
142         * menu.
143         *
144         * @param widget the widget to be added.
145         */
146        protected void addWidget(Widget widget) {
147            super.add(widget);
148        }
149    
150        private native void configure(Element e) /*-{
151            $wnd.jQuery(e).dropdown();
152        }-*/;
153    
154        /**
155         * Pull the dropdown menu to right
156         *
157         * @param rightDropdown <code>true</code> pull to right, otherwise to left. Default is <code>false</code>
158         */
159        public void setRightDropdown(boolean rightDropdown) {
160            if (rightDropdown) {
161                menu.addStyleName("pull-right");
162            } else {
163                menu.removeStyleName("pull-right");
164            }
165        }
166    
167        public HandlerRegistration addChangeHandler(ChangeHandler handler) {
168            return addDomHandler(handler, ChangeEvent.getType());
169        }
170    
171        private class NavLinkClickHandler implements ClickHandler {
172    
173            @Override
174            public void onClick(ClickEvent event) {
175                link = (NavLink) event.getSource();
176                DomEvent.fireNativeEvent(Document.get().createChangeEvent(), DropdownBase.this);
177            }
178    
179        }
180    
181        /**
182         * Method to get the {@link NavLink} that has been clicked most recently.
183         *
184         * @return Last clicked NavLink or <code>null</code> if none have been clicked.
185         */
186        public NavLink getLastSelectedNavLink() {
187            return link;
188        }
189    
190        @Override
191        public void clear() {
192            menu.clear();
193        }
194    }
