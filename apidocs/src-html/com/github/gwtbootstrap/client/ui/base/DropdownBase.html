016    package com.github.gwtbootstrap.client.ui.base;
017    
018    import com.github.gwtbootstrap.client.ui.Divider;
019    import com.github.gwtbootstrap.client.ui.NavLink;
020    import com.github.gwtbootstrap.client.ui.NavText;
021    import com.github.gwtbootstrap.client.ui.UnorderedList;
022    import com.github.gwtbootstrap.client.ui.constants.Constants;
023    import com.google.gwt.dom.client.Element;
024    import com.google.gwt.user.client.ui.Widget;
025    
026    //@formatter:off
027    /**
028     * Base class for dropdown widgets.
029     * 
030     * @since 2.0.2.0
031     * 
032     * @author Carlos A Becker
033     * @author Dominik Mayer
034     * 
035     */
036    //@formatter:on
037    public abstract class DropdownBase extends ComplexWidget {
038    
039            private UnorderedList menu = new UnorderedList();
040    
041            private IconAnchor trigger;
042    
043            private NavLink link;
044    
045            /**
046             * Creates a new widget of the given type.
047             * 
048             * @param type
049             *            the HTML tag to be used for the widget
050             */
051            public DropdownBase(String type) {
052                    super(type);
053                    createAndAddTrigger();
054                    menu.setStyleName("dropdown-menu");
055                    super.add(menu);
056            }
057    
058            private void createAndAddTrigger() {
059                    trigger = createTrigger();
060                    trigger.addStyleName("dropdown-toggle");
061                    trigger.getElement().setAttribute(Constants.DATA_TOGGLE, "dropdown");
062                    super.add(trigger);
063            }
064    
065            /**
066             * Sets the text of the dropdown trigger.
067             * 
068             * @param text
069             */
070            public void setText(String text) {
071                    trigger.setText(text);
072            }
073    
074            /**
075             * Implement this method to create the trigger appropriate for your widget.
076             * It has to be an {@link IconAnchor} or a subtype.
077             * 
078             * @return the created trigger
079             */
080            protected abstract IconAnchor createTrigger();
081    
082            public void setDropup(boolean dropup) {
083                    if (dropup)
084                            addStyleName(Constants.DROPUP);
085                    else
086                            removeStyleName(Constants.DROPUP);
087            }
088    
089            /**
090             * {@inheritDoc}
091             */
092            @Override
093            protected void onLoad() {
094                    super.onLoad();
095                    if (trigger != null) {
096                            configure(trigger.getElement());
097                    }
098            }
099    
100            /**
101             * Adds a widget to the dropdown menu.
102             * 
103             * @param widget
104             *            the widget that will be added to the menu
105             * 
106             * @see #addWidget(Widget)
107             */
108            @Override
109            public void add(Widget widget) {
110                    if (canBeAdded(widget))
111                            menu.add(widget);
112                    else
113                            throw new IllegalArgumentException("Only NavLink, NavText and "
114                                            + "Divider can be added to the Dropdown");
115            }
116    
117            /**
118             * Checks whether a widget can be added to the dropdown.
119             * <p>
120             * By default, the dropdown accepts {@link NavLink}, {@link NavText} and
121             * {@link Divider} widgets. Override this class to allow more widgets.
122             * 
123             * @param widget
124             *            the widget that should be added
125             * 
126             * @return <code>true</code> if the widget can be added to the Dropdown
127             */
128            protected boolean canBeAdded(Widget widget) {
129                    return ((widget instanceof NavLink) || (widget instanceof NavText) || (widget instanceof Divider));
130            }
131    
132            /**
133             * Adds a widget to the the dropdown widget, <b>not</b> to the dropdown
134             * menu.
135             * 
136             * <p>
137             * Use {@link #add(Widget)} if you want to add a widget to the dropdown
138             * menu.
139             * 
140             * @param widget
141             *            the widget to be added.
142             */
143            protected void addWidget(Widget widget) {
144                    super.add(widget);
145            }
146    
147            private native void configure(Element e) /*-{
148                    $wnd.jQuery(e).dropdown();
149            }-*/;
150    }
