016package com.github.gwtbootstrap.client.ui.base;
017
018import com.github.gwtbootstrap.client.ui.Icon;
019import com.github.gwtbootstrap.client.ui.constants.IconType;
020import com.google.gwt.user.client.ui.HasText;
021
022/**
023 * An Anchor with optional image and caret.
024 * 
025 * <p>
026 * It uses a HTML {@code <a>} tag and can contain text and child widgets. But
027 * not both at the same time.
028 * </p>
029 * 
030 * <p>
031 * <h3>UiBinder Usage:</h3>
032 * {@code <b:IconAnchor icon="plane" href="www.twitter.com">Some Text</b:IconAnchor>}
033 * </p>
034 * 
035 * <p>
036 * Here we add a second Icon:
037 * 
038 * <pre>
039 * {@code <b:IconAnchor icon="star" text="There is a widget so the text goes here">
040 *     <b:Icon type="star" />
041 * </b:IconAnchor>}
042 * </pre>
043 * 
044 * All parameter are optional. All setters can be used as parameters.
045 * </p>
046 * 
047 * @since 2.0.2.0
048 * 
049 * @author Dominik Mayer
050 * 
051 */
052public class IconAnchor extends ComplexWidget implements HasText, HasIcon {
053
054        private Icon icon = new Icon();
055
056        private InlineLabel label = new InlineLabel();
057
058        private String text = "";
059
060        private Caret caret = new Caret(false);
061
062        /**
063         * Creates the widget and sets the {@code href} property to
064         * {@code javascript:;} in order to avoid problems when clicking on it.
065         */
066        public IconAnchor() {
067                super("a");
068                super.add(icon);
069                super.add(label);
070                super.add(caret);
071                setHref("javascript:;");
072        }
073
074        /**
075         * {@inheritDoc}
076         */
077        public void setIcon(IconType type) {
078                if (type != null)
079                        this.icon.setType(type);
080        }
081
082        /**
083         * {@inheritDoc}
084         * 
085         * @deprecated This method should never be called directly. It will break
086         *             your implementation if any style names change. The only valid
087         *             use is inside UiBinder files where it processes the
088         *             <code>icon="..."</code> argument. Use
089         *             {@link #setIcon(IconType)} instead!
090         */
091        @Deprecated
092        public void setIcon(String iconName) {
093                icon.setType(iconName);
094        }
095
096        /**
097         * {@inheritDoc}
098         */
099        public void setIconColor(Icon.Color color) {
100                icon.setColor(color);
101        }
102
103        /**
104         * {@inheritDoc}
105         * 
106         * @deprecated This method should never be called directly. It will break
107         *             your implementation if any style names change. The only valid
108         *             use is inside UiBinder files where it processes the
109         *             <code>icon="..."</code> argument. Use
110         *             {@link #setIconColor(Icon.Color)} instead!
111         */
112        @Deprecated
113        public void setIconColor(String color) {
114                icon.setColor(color);
115        }
116
117        /**
118         * {@inheritDoc}
119         */
120        public void setText(String text) {
121                this.text = text;
122                label.setText(" " + text + " ");
123        }
124
125        /**
126         * {@inheritDoc}
127         */
128        public String getText() {
129                return text;
130        }
131
132        /**
133         * Sets the <code>href</code> property of the anchor tag.
134         * <p>
135         * Use it only to set links to another page. Links on the same page or GWT
136         * tokens should be set with {@code #setTargetHistoryToken(String)}
137         * 
138         * @param href
139         *            the website URL
140         */
141        public void setHref(String href) {
142                getElement().setAttribute("href", href);
143        }
144
145        /**
146         * @return the URL of the website this link is pointing to
147         */
148        public String getHref() {
149                return getElement().getAttribute("href");
150        }
151
152        /**
153         * Shows or hides the caret.
154         * 
155         * @param visible
156         *            <code>true</code> if the caret should be shown.
157         */
158        public void setCaret(boolean visible) {
159                caret.setVisible(visible);
160        }
161
162        /**
163         * Sets the <code>href</code> property of the anchor tag to
164         * <code>"#" + targetHistoryToken</code>.
165         * 
166         * @param targetHistoryToken
167         *            the history token
168         */
169        public void setTargetHistoryToken(String targetHistoryToken) {
170                setHref("#" + targetHistoryToken);
171        }
172
173}
