001    package com.github.gwtbootstrap.client.ui.base;
002    
003    import com.github.gwtbootstrap.client.ui.constants.Constants;
004    import com.google.gwt.dom.client.Element;
005    import com.google.gwt.user.client.ui.Widget;
006    
007    
008    /**
009     * The hepper class for {@link IsSearchQuery}.
010     * 
011     * @since 2.0.3.0
012     * @author ohashi keisuke
013     */
014    public class SearchQueryStyleHelper {
015    
016            /**
017             * Set search-query style to the element.
018             * @param widget applied style widget
019             * @param searchQuery true:add search-query css-class/false:remove search-query css-class
020             */
021            public static void setSearchQuery(Widget widget , boolean searchQuery) {
022                    setSearchQuery(widget.getElement(), searchQuery);
023            }
024            
025            /**
026             * is the element search-query style?
027             * @param widget target widget
028             * @return true:has search-query css-class/false:has no search-query cass-class.
029             */
030            public static boolean isSearchQuery(Widget widget) {
031                    return isSearchQuery(widget.getElement());
032            }
033    
034            /**
035             * Set search-query style to the element.
036             * @param elem the applied style element
037             * @param searchQuery true:add search-query css-class/false:remove search-query css-class
038             */
039            public static void setSearchQuery(Element elem, boolean searchQuery) {
040                    if(isSearchQuery(elem)) {
041                            elem.removeClassName(Constants.SEARCH_QUERY);
042                    }
043                    
044                    if(searchQuery) {
045                            elem.addClassName(Constants.SEARCH_QUERY);
046                    }
047            }
048            
049            /**
050             * is the element search-query style?
051             * @param elem target element.
052             * @return true:has search-query css-class/false:has no search-query cass-class.
053             */
054            public static boolean isSearchQuery(Element elem) {
055                    return elem.getClassName().contains(Constants.SEARCH_QUERY);
056            }
057    
058    }
