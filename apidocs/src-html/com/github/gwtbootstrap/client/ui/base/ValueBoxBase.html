001package com.github.gwtbootstrap.client.ui.base;
002
003import com.github.gwtbootstrap.client.ui.constants.AlternateSize;
004import com.github.gwtbootstrap.client.ui.constants.Constants;
005import com.google.gwt.core.client.GWT;
006import com.google.gwt.dom.client.Element;
007import com.google.gwt.text.shared.Parser;
008import com.google.gwt.text.shared.Renderer;
009
010/**
011 * A ValueBoxBase extending for Bootstarp style.
012 * 
013 * <pre>
014 * It is GWT's {@link com.google.gwt.user.client.ui.ValueBoxBase} extending for Bootstrap style.
015 * 
016 * @since 2.0.3.0
017 * @author ohashi keisuke
018 * @see com.google.gwt.user.client.ui.ValueBoxBase
019 * @param <T> the value type
020 */
021public class ValueBoxBase<T> extends com.google.gwt.user.client.ui.ValueBoxBase<T> implements HasPlaceholder, HasAlternateSize, IsSearchQuery, HasSize, HasId {
022
023        /** placeholderHelper */
024        private PlaceholderHelper placeholderHelper = GWT.create(PlaceholderHelper.class);
025
026        /**
027         * Creates a value box that wraps the given browser element handle. This is
028         * only used by subclasses.
029         * 
030         * @param elem
031         *            the browser element to wrap
032         */
033        protected ValueBoxBase(Element elem,
034                Renderer<T> renderer,
035                Parser<T> parser) {
036                super(elem, renderer, parser);
037        }
038
039        /**
040         * {@inheritDoc}
041         */
042        @Override
043        public void setPlaceholder(String placeholder) {
044                placeholderHelper.setPlaceholer(getElement(), placeholder);
045        }
046
047        /**
048         * {@inheritDoc}
049         */
050        @Override
051        public String getPlaceholder() {
052                return placeholderHelper.getPlaceholder(getElement());
053        }
054
055        /**
056         * {@inheritDoc}
057         */
058        @Override
059        public void setSearchQuery(boolean searchQuery) {
060                SearchQueryStyleHelper.setSearchQuery(this, searchQuery);
061        }
062
063        /**
064         * {@inheritDoc}
065         */
066        @Override
067        public boolean isSearchQuery() {
068                return SearchQueryStyleHelper.isSearchQuery(this);
069        }
070
071        /**
072         * {@inheritDoc}
073         */
074        @Override
075        public void setAlternateSize(AlternateSize size) {
076                StyleHelper.changeStyle(this, size, AlternateSize.class);
077        }
078
079        /**
080         * {@inheritDoc}
081         */
082        @Override
083        public void setSize(int size) {
084                SizeHelper.setSize(this, size);
085        }
086
087        /**
088         * {@inheritDoc}
089         */
090        @Override
091        public String getId() {
092                return getElement().getId();
093        }
094
095        /**
096         * {@inheritDoc}
097         */
098        @Override
099        public void setId(String id) {
100                getElement().setId(id);
101        }
102
103        /**
104         * {@inheritDoc}
105         */
106        @Override
107        public void setEnabled(boolean enabled) {
108                super.setEnabled(enabled);
109                if (enabled) {
110                        removeStyleName(Constants.DISABLED);
111                } else {
112                        addStyleName(Constants.DISABLED);
113                }
114        }
115}
