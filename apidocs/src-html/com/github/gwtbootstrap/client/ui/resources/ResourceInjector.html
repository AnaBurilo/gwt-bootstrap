016    package com.github.gwtbootstrap.client.ui.resources;
017    
018    import com.google.gwt.core.client.GWT;
019    import com.google.gwt.dom.client.StyleInjector;
020    import com.google.gwt.resources.client.TextResource;
021    
022    /**
023     * Utility class to inject our resources into modules page. Use it to inject
024     * JavaScript and CSS files.
025     * 
026     * @author Carlos Alexandro Becker
027     * @since 21/01/2012
028     */
029    public class ResourceInjector {
030    
031            private static final ResourceAdapter ADAPTER = GWT.create(ResourceAdapter.class);
032    
033            /**
034             * Import the required styles and JSs to the head of modules HTML.
035             */
036            public static void configure() {
037                    
038                    Resources res = ADAPTER.getResources();
039                    
040                    injectCss(res.bootstrapCss());
041                    if (ADAPTER.hasResponsiveDesign())
042                            activateResponsiveDesign(res);
043                    injectJs(res.jquery());
044                    injectJs(res.bootstrapJs());
045    //              injectJs(res.bootstrapCollapseJs());
046            }
047    
048            public static void injectCss(TextResource r) {
049    //              r.ensureInjected();
050                    StyleInjector.inject(r.getText());
051            }
052    
053            public static void injectJs(TextResource r) {
054                    JavaScriptInjector.inject(r.getText());
055            }
056            
057            private static void activateResponsiveDesign(Resources res) {
058                    injectCss(res.bootstrapResponsiveCss());
059                    MetaInjector.inject("viewport", "width=device-width, initial-scale=1.0");
060            }
061            
062    
063            
064    }
