016    package com.github.gwtbootstrap.client.ui.resources;
017    
018    import com.github.gwtbootstrap.client.ui.base.IsResponsive;
019    
020    /**
021     * <h3>Using custom css/js resources.</h3>
022     * 
023     * <p>
024     * If you need to adapt resources, implement this interface, and add a
025     * <code>replace-with</code> tag to your module xml. Example:
026     * </p>
027     * 
028     * <p>
029     * 1. Create a Resources Interface (extending {@link Resources}) override the
030     * getters of the files you want to replace.
031     * 
032     * <pre>
033     *      public interface MyResources extends Resources {
034     *              {@literal @Source("mycss/custom-bootstrap.min.css")}
035     *              TextResource bootstrapCss();
036     *      }
037     * </pre>
038     * 
039     * </p>
040     * 
041     * <p>
042     * 2. Create a <code>ResourceAdapter</code>.
043     * 
044     * <pre>
045     *      public MyResourceAdapter implements ResourceAdapter {
046     *              public Resources getResources() {
047     *                      return GWT.create(MyResources.class);
048     *              }
049     *      }
050     * </pre>
051     * 
052     * </p>
053     * 
054     * <p>
055     * 3. Add a <code>replace-with</code> tag to your module xml (
056     * <code>*.gwt.xml</code>).
057     * 
058     * <pre>
059     * {@literal <replace-with class="userpackage.MyResourceAdapter">
060     *       <when-type-is class="com.github.gwtbootstrap.client.ui.resources.ResourceAdapter" />
061     * </replace-with>}
062     * </pre>
063     * 
064     * </p>
065     * 
066     * @since 2.0.2.0
067     * 
068     * @author soundTricker
069     * @author Dominik Mayer
070     * 
071     * @see Resources
072     * @see ResourceAdapterImpl
073     */
074    public interface ResourceAdapter {
075    
076            /**
077             * Get the Bootstrap Resources that should be used for this project.
078             * 
079             * @return the Bootstrap Resources
080             */
081            Resources getResources();
082    
083            //@formatter:off
084            /**
085             * Determines whether the project uses a responsive design.
086             * 
087             * <p>
088             * If the responsive
089             * design is enabled, the interface can adapt to the screen size and show
090             * different content on smartphones, tablets and desktop computers.
091             * </p>
092             * 
093             * @return <code>true</code> if the responsive features should be enabled.
094             * Default: <code>false</code>
095             * 
096             * @see IsResponsive
097             * 
098             * @see <a href="http://twitter.github.com/bootstrap/scaffolding.html#responsive">Bootstrap documentation</a>
099             */
100            boolean hasResponsiveDesign();
101    }
