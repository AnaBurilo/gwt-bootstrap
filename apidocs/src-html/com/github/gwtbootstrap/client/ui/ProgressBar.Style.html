016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.base.DivWidget;
019import com.github.gwtbootstrap.client.ui.base.StyleHelper;
020import com.github.gwtbootstrap.client.ui.constants.Constants;
021import com.github.gwtbootstrap.client.ui.resources.Bootstrap;
022import com.google.gwt.dom.client.Style.Unit;
023
024//@formatter:off
025/**
026 * The progress bar.
027 * 
028 * @since 2.0.4.0
029 * @author Dominik Mayer
030 */
031//@formatter:on
032public class ProgressBar extends DivWidget {
033    
034        public enum Style implements com.github.gwtbootstrap.client.ui.base.Style {
035
036                DEFAULT(""),
037
038                STRIPED(Bootstrap.progress_striped),
039                
040                ANIMATED(Bootstrap.progress_striped);           
041                
042                private final String styleName;
043
044        private Style(String styleName) {
045            this.styleName = styleName;
046                    
047                }
048
049                public String get() {
050                    return this.styleName;
051                }
052        }
053        
054        public enum Color implements com.github.gwtbootstrap.client.ui.base.Style {
055        DEFAULT(""),
056            INFO("progress-info"),
057        SUCCESS("progress-success"),
058        DANGER("progress-danger"),
059        WARNING("progress-warning")
060        ;
061        
062        private final String styleName;
063
064        private Color(String styleName) {
065            this.styleName = styleName;
066        }
067
068        @Override
069        public String get() {
070            return this.styleName;
071        }
072            
073        }
074
075        private DivWidget bar = new DivWidget();
076
077        public ProgressBar() {
078            setStylePrimaryName(Bootstrap.progress);
079                addStyleName(Bootstrap.progress);
080                bar.addStyleName(Bootstrap.bar);
081                setColor(Color.DEFAULT);
082                add(bar);
083        }
084
085        public ProgressBar(Style style) {
086                this();
087                setType(style);
088        }
089
090        public void setType(Style style) {
091            StyleHelper.changeStyle(this, style, Style.class);
092            setActive(Style.ANIMATED == style);
093        }
094
095        public void setPercent(int percent) {
096                bar.getElement().getStyle().setWidth(percent, Unit.PCT);
097        }
098
099        public int getPercent() {
100                String width = bar.getElement().getStyle().getWidth();
101                if (width == null)
102                        return 0;
103                else
104                        return Integer.valueOf(width.substring(0, width.indexOf("%")));
105        }
106        
107        /**
108         * Set progress bar color
109         * @param color color
110         */
111        public void setColor(Color color) {
112            StyleHelper.changeStyle(this, color, Color.class);
113        }
114        
115        /**
116         * Set active style.
117         * <p>
118         * if set type {@link Style#STRIPED} and this set true, bar is animated
119         * @param active true:active false:deactive
120         */
121        public void setActive(boolean active) {
122            setStyleName(Constants.ACTIVE , active);
123        }
124
125}
