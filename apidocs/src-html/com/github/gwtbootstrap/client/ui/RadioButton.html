014/**
015 * RadioButton widgets.
016 * <p>
017 * Re-design for Bootstrap.
018 * </p>
019 * 
020 * @since 2.0.3.0
021 * @author ohashi keisuke
022 */
023public class RadioButton extends CheckBox {
024
025        public static final DirectionEstimator DEFAULT_DIRECTION_ESTIMATOR = DirectionalTextHelper.DEFAULT_DIRECTION_ESTIMATOR;
026
027        private Boolean oldValue;
028
029        /**
030         * Creates a new radio associated with a particular group name. All radio
031         * buttons associated with the same group name belong to a
032         * mutually-exclusive set.
033         * 
034         * Radio buttons are grouped by their name attribute, so changing their name
035         * using the setName() method will also change their associated group.
036         * 
037         * @param name
038         *            the group name with which to associate the radio button
039         */
040        @UiConstructor
041        public RadioButton(String name) {
042                super(DOM.createInputRadio(name));
043
044                sinkEvents(Event.ONCLICK);
045                sinkEvents(Event.ONMOUSEUP);
046                sinkEvents(Event.ONBLUR);
047                sinkEvents(Event.ONKEYDOWN);
048        }
049
050        /**
051         * Creates a new radio associated with a particular group, and initialized
052         * with the given HTML label. All radio buttons associated with the same
053         * group name belong to a mutually-exclusive set.
054         * 
055         * Radio buttons are grouped by their name attribute, so changing their name
056         * using the setName() method will also change their associated group.
057         * 
058         * @param name
059         *            the group name with which to associate the radio button
060         * @param label
061         *            this radio button's html label
062         */
063        public RadioButton(String name,
064                SafeHtml label) {
065                this(name, label.asString(), true);
066        }
067
068        /**
069         * @see #RadioButton(String, SafeHtml)
070         * 
071         * @param name
072         *            the group name with which to associate the radio button
073         * @param label
074         *            this radio button's html label
075         * @param dir
076         *            the text's direction. Note that {@code DEFAULT} means
077         *            direction should be inherited from the widget's parent
078         *            element.
079         */
080        public RadioButton(String name,
081                SafeHtml label,
082                Direction dir) {
083                this(name);
084                setHTML(label, dir);
085        }
086
087        /**
088         * @see #RadioButton(String, SafeHtml)
089         * 
090         * @param name
091         *            the group name with which to associate the radio button
092         * @param label
093         *            this radio button's html label
094         * @param directionEstimator
095         *            A DirectionEstimator object used for automatic direction
096         *            adjustment. For convenience,
097         *            {@link #DEFAULT_DIRECTION_ESTIMATOR} can be used.
098         */
099        public RadioButton(String name,
100                SafeHtml label,
101                DirectionEstimator directionEstimator) {
102                this(name);
103                setDirectionEstimator(directionEstimator);
104                setHTML(label.asString());
105        }
106

113         * using the setName() method will also change their associated group.
114         * 
115         * @param name
116         *            the group name with which to associate the radio button
117         * @param label
118         *            this radio button's label
119         */
120        public RadioButton(String name,
121                String label) {
122                this(name);
123                setText(label);
124        }
125
126        /**
127         * @see #RadioButton(String, SafeHtml)
128         * 
129         * @param name
130         *            the group name with which to associate the radio button
131         * @param label
132         *            this radio button's label
133         * @param dir
134         *            the text's direction. Note that {@code DEFAULT} means
135         *            direction should be inherited from the widget's parent
136         *            element.
137         */
138        public RadioButton(String name,
139                String label,
140                Direction dir) {
141                this(name);
142                setText(label, dir);
143        }

145        /**
146         * @see #RadioButton(String, SafeHtml)
147         * 
148         * @param name
149         *            the group name with which to associate the radio button
150         * @param label
151         *            this radio button's label
152         * @param directionEstimator
153         *            A DirectionEstimator object used for automatic direction
154         *            adjustment. For convenience,
155         *            {@link #DEFAULT_DIRECTION_ESTIMATOR} can be used.
156         */
157        public RadioButton(String name,
158                String label,
159                DirectionEstimator directionEstimator) {
160                this(name);
161                setDirectionEstimator(directionEstimator);
162                setText(label);
163        }
164
165        /**
166         * Creates a new radio button associated with a particular group, and
167         * initialized with the given label (optionally treated as HTML). All radio
168         * buttons associated with the same group name belong to a
169         * mutually-exclusive set.
170         * 
171         * Radio buttons are grouped by their name attribute, so changing their name
172         * using the setName() method will also change their associated group.
173         * 
174         * @param name
175         *            name the group with which to associate the radio button
176         * @param label
177         *            this radio button's label
178         * @param asHTML
179         *            <code>true</code> to treat the specified label as HTML
180         */
181        public RadioButton(String name,
182                String label,
183                boolean asHTML) {
184                this(name);
185                if (asHTML) {
186                        setHTML(label);
187                } else {
188                        setText(label);
189                }
190        }
191
192        /**
193         * Overridden to send ValueChangeEvents only when appropriate.
194         */
195        @Override
196        public void onBrowserEvent(Event event) {
197                switch (DOM.eventGetType(event)) {
198                case Event.ONMOUSEUP:
199                case Event.ONBLUR:
200                case Event.ONKEYDOWN:
201                        // Note the old value for onValueChange purposes (in ONCLICK case)
202                        oldValue = getValue();
203                        break;
204
205                case Event.ONCLICK:
206                        EventTarget target = event.getEventTarget();
207                        if (Element.is(target) && asLabel().isOrHasChild(Element.as(target))) {
208
209                                // They clicked the label. Note our pre-click value, and
210                                // short circuit event routing so that other click handlers
211                                // don't hear about it
212                                oldValue = getValue();
213                                return;
214                        }
215
216                        // It's not the label. Let our handlers hear about the
217                        // click...
218                        super.onBrowserEvent(event);
219                        // ...and now maybe tell them about the change
220                        ValueChangeEvent.fireIfNotEqual(RadioButton.this, oldValue, getValue());
221                        return;
222                }
223
224                super.onBrowserEvent(event);
225        }
226
227        /**
228         * Change the group name of this radio button.
229         * 
230         * Radio buttons are grouped by their name attribute, so changing their name
231         * using the setName() method will also change their associated group.
232         * 
233         * If changing this group name results in a new radio group with multiple
234         * radio buttons selected, this radio button will remain selected and the
235         * other radio buttons will be unselected.
236         * 
237         * @param name
238         *            name the group with which to associate the radio button
239         */
240        @Override
241        public void setName(String name) {
242                // Just changing the radio button name tends to break groupiness,
243                // so we have to replace it. Note that replaceInputElement is careful
244                // not to propagate name when it propagates everything else
245                replaceInputElement(DOM.createInputRadio(name));
246        }
247
248        @Override
249        public void sinkEvents(int eventBitsToAdd) {
250                // Like CheckBox, we want to hear about inputElem. We
251                // also want to know what's going on with the label, to
252                // make sure onBrowserEvent is able to record value changes
253                // initiated by label events
254                if (isOrWasAttached()) {
255                        Event.sinkEvents(inputElem, eventBitsToAdd | Event.getEventsSunk(inputElem));
256                        Event.sinkEvents(asLabel(), eventBitsToAdd | Event.getEventsSunk(asLabel()));
257                } else {
258                        super.sinkEvents(eventBitsToAdd);
259                }
260        }
261
262        /**
263         * No-op. CheckBox's click handler is no good for radio button, so don't use
264         * it. Our event handling is all done in {@link #onBrowserEvent}
265         */
266        @Override
267        protected void ensureDomEventHandlers() {
268        }
