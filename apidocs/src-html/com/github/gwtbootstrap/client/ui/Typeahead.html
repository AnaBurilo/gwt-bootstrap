001package com.github.gwtbootstrap.client.ui;
002
003import java.util.Collection;
004
005import com.github.gwtbootstrap.client.ui.base.MarkupWidget;
006import com.github.gwtbootstrap.client.ui.base.TextBoxBase;
007import com.google.gwt.core.client.JavaScriptObject;
008import com.google.gwt.core.client.JsArrayString;
009import com.google.gwt.core.client.Scheduler;
010import com.google.gwt.core.client.Scheduler.ScheduledCommand;
011import com.google.gwt.user.client.Element;
012import com.google.gwt.user.client.ui.MultiWordSuggestOracle;
013import com.google.gwt.user.client.ui.SuggestOracle;
014import com.google.gwt.user.client.ui.SuggestOracle.Callback;
015import com.google.gwt.user.client.ui.SuggestOracle.Request;
016import com.google.gwt.user.client.ui.SuggestOracle.Response;
017import com.google.gwt.user.client.ui.SuggestOracle.Suggestion;
018import com.google.gwt.user.client.ui.Widget;
019
020public class Typeahead extends MarkupWidget {
021    
022    private int displayItems = 8;
023    
024    private int minLength = 1;
025
026    private final SuggestOracle oracle;
027    
028    /**
029     * Constructor for {@link Typeahead}. Creates a
030     * {@link MultiWordSuggestOracle} to use with this
031     */
032    public Typeahead() {
033        this(new MultiWordSuggestOracle());
034    }
035    
036    /**
037     * Constructor for {@link Typeahead}.
038     * 
039     * @param oracle the oracle for this <code>Typeahead</code>
040     */
041    public Typeahead(SuggestOracle oracle) {
042        this.oracle = oracle;
043    }
044
045    /**
046     * {@inheritDoc}
047     */
048    @Override
049    public void setWidget(Widget w) {
050        
051        if(!(w instanceof TextBoxBase || w instanceof com.google.gwt.user.client.ui.TextBoxBase)) {
052            throw new IllegalArgumentException("Typeahead should be set TextBoxBase childs");
053        }
054        
055        super.setWidget(w);
056    }
057    
058    @Override
059    public Widget asWidget() {
060        
061        if(widget != null) {
062            Scheduler.get().scheduleDeferred(new ScheduledCommand() {
063                
064                @Override
065                public void execute() {
066                    reconfigure();
067                }
068            });
069            
070        }
071        
072        return super.asWidget();
073    }
074    
075    /**
076     * reconfigure setting.
077     */
078    public void reconfigure() {
079        
080        if(widget == null) return;
081        
082        removeDataIfExists(widget.getElement());
083        
084        configure(widget.getElement(), displayItems , minLength);
085    }
086    
087    /**
088     * Get the max number of items to display in the dropdown.
089     * @return  the max number of items to display in the dropdown
090     */
091    public int getDisplayItemCount() {
092        return displayItems;
093    }
094
095    /**
096     * Set max number of items to display in the dropdown.
097     * @param displayItems the max number of items to display in the dropdown
098     */
099    public void setDisplayItemCount(int displayItems) {
100        this.displayItems = displayItems;
101    }
102
103    /**
104     * Get the minimum character length needed before triggering autocomplete suggestions.
105     * @return the minimum character length needed before triggering autocomplete suggestions
106     */
107    public int getMinLength() {
108        return minLength;
109    }
110
111    /**
112     * Set the minimum character length needed before triggering autocomplete suggestions.
113     * @param minLength The minimum character length needed before triggering autocomplete suggestions.
114     */
115    public void setMinLength(int minLength) {
116        this.minLength = minLength;
117    }
118
119    /**
120     * Get suggest oracle
121     * @return oracle
122     */
123    public SuggestOracle getSuggestOracle() {
124        return oracle;
125    }
126
127    private void query(String query ,final JavaScriptObject process) {
128        Callback callback = new Callback() {
129            
130            @Override
131            public void onSuggestionsReady(Request request, Response response) {
132                callback(process, response);
133            }
134        };
135        
136        if(query != null && !query.isEmpty()) {
137            oracle.requestSuggestions(new Request(query, displayItems), callback);
138        } else {
139            oracle.requestDefaultSuggestions(new Request(), callback);
140        }
141    }
142    
143    private void callback(final JavaScriptObject process, Response response) {
144        Collection<? extends Suggestion> suggestions = response.getSuggestions();
145        
146        JsArrayString jsArrayString = JavaScriptObject.createArray().cast();
147        
148        for (Suggestion suggestion : suggestions) {
149            jsArrayString.push(suggestion.getReplacementString());
150        }
151        callProcess(jsArrayString, process);
152    }
153    
154    //@formatter:off
155    private native void callProcess(JsArrayString items ,JavaScriptObject process) /*-{
156        process(items);
157    }-*/;
158
159    private native void removeDataIfExists(Element element) /*-{
160        var $this = $wnd.jQuery(element);
161        $this.data("typeahead") && $this.removeData("typeahead");
162    }-*/;
163    
164    public native void configure(Element element,int items, int minLength) /*-{
165        var that = this;
166        $wnd.jQuery(element).typeahead({
167            "source" : function(query , process) {
168                that.@com.github.gwtbootstrap.client.ui.Typeahead::query(Ljava/lang/String;Lcom/google/gwt/core/client/JavaScriptObject;)(query , process);
169            },
170            "items" : items,
171            "minLength" : minLength
172        });
173    }-*/;
174    //@formatter:on
175}
