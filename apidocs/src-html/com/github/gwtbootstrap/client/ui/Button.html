087    /**
088     * Bootstrap Button Component.
089     * 
090     * @author Carlos A Becker
091     */
092    public class Button extends IconAnchor implements HasClickHandlers,
093                    HasDoubleClickHandlers, HasEnabled, HasAllDragAndDropHandlers,
094                    HasAllFocusHandlers, HasAllGestureHandlers, HasAllKeyHandlers,
095                    HasAllMouseHandlers, HasAllTouchHandlers {
096    
097            {
098                    removeStyleName("gwt-Button");
099                    addStyle(Bootstrap.Button.BTN);
100            }
101    
102            public Button() {
103                    super();
104                    getElement().appendChild(Document.get().createElement("i"));
105            }
106    
107            public Button(String text) {
108                    this();
109                    setText(text);
110            }
111    
112            public Button(String text, IconType icon) {
113                    this(text);
114                    setIcon(icon);
115            }
116    
117            public void setStyle(Bootstrap.Button.Type type) {
118                    for (Bootstrap.Button.Type t : Bootstrap.Button.Type.values())
119                            removeStyle(t);
120                    addStyle(type);
121            }
122    
123            public void setType(String typeString) {
124                    String type = "btn-" + typeString;
125    
126                    for (Bootstrap.Button.Type t : Bootstrap.Button.Type.values()) {
127                            if (type.equals(t.get()))
128                                    addStyle(t);
129                            else
130                                    removeStyle(t);
131                    }
132            }
133    
134            public void setSize(Bootstrap.Button.Size size) {
135                    for (Bootstrap.Button.Size s : Bootstrap.Button.Size.values())
136                            removeStyle(s);
137                    addStyle(size);
138            }
139    
140            public void setSize(String sizeString) {
141                    String size = "btn-" + sizeString;
142    
143                    for (Bootstrap.Button.Size s : Bootstrap.Button.Size.values()) {
144                            if (size.equals(s.get()))
145                                    addStyle(s);
146                            else
147                                    removeStyle(s);
148                    }
149            }
150    
151            public boolean isEnabled() {
152                    return getStyleName().contains("disabled");
153            }
154    
155            public void setEnabled(boolean enabled) {
156                    if (enabled)
157                            removeStyleName(Bootstrap.disabled);
158                    else
159                            addStyleName(Bootstrap.disabled);
160            }
161    
162            public HandlerRegistration addClickHandler(ClickHandler handler) {
163                    return addDomHandler(handler, ClickEvent.getType());
164            }

167                            addDoubleClickHandler(DoubleClickHandler handler) {
168                    return addDomHandler(handler, DoubleClickEvent.getType());
