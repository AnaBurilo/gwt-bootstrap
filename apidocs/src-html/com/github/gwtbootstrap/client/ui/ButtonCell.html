001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.constants.ButtonType;
004import com.github.gwtbootstrap.client.ui.constants.IconType;
005import com.google.gwt.safehtml.shared.SafeHtml;
006import com.google.gwt.safehtml.shared.SafeHtmlBuilder;
007
008/**
009 * A ButtonCell with Bootstrap Style, supporting Icons and icon types.
010 * 
011 * @author Carlos A Becker
012 * @since 2.0.2.0
013 * 
014 * @see Button
015 * 
016 */
017public class ButtonCell extends com.google.gwt.cell.client.ButtonCell {
018
019        private IconType icon = null;
020        private ButtonType type = null;
021
022        public ButtonCell() {
023        }
024
025        public ButtonCell(ButtonType type) {
026                super();
027                this.type = type;
028        }
029
030        public ButtonCell(IconType icon) {
031                super();
032                this.icon = icon;
033        }
034
035        public ButtonCell(IconType icon, ButtonType type) {
036                super();
037                this.icon = icon;
038                this.type = type;
039        }
040
041        @Override
042        public void render(Context context, SafeHtml data, SafeHtmlBuilder sb) {
043                sb.appendHtmlConstant("<button type=\"button\" class=\"btn "
044                                + (type != null ? type.get() : "") + "\" tabindex=\"-1\">");
045                if (data != null) {
046                        if (icon != null)
047                                sb.appendHtmlConstant("<i class=\"" + icon.get() + "\"></i> ");
048                        sb.append(data);
049                }
050                sb.appendHtmlConstant("</button>");
051        }
052
053}
