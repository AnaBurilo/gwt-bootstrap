001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.constants.ButtonType;
004import com.github.gwtbootstrap.client.ui.constants.IconColor;
005import com.github.gwtbootstrap.client.ui.constants.IconType;
006import com.google.gwt.safehtml.shared.SafeHtml;
007import com.google.gwt.safehtml.shared.SafeHtmlBuilder;
008
009/**
010 * A ButtonCell with Bootstrap Style, supporting Icons and icon types.
011 * 
012 * @author Carlos A Becker
013 * @since 2.0.3.0
014 * 
015 * @see Button
016 * 
017 */
018public class ButtonCell extends com.google.gwt.cell.client.ButtonCell {
019
020        private IconType icon = null;
021        private ButtonType type = null;
022    private String color = null;
023
024        public ButtonCell() {
025        }
026
027        public ButtonCell(ButtonType type) {
028                super();
029                this.type = type;
030        }
031
032        public ButtonCell(IconType icon) {
033                super();
034                this.icon = icon;
035        }
036
037    public ButtonCell(String color) {
038        this.color = color;
039    }
040
041    public ButtonCell(ButtonType type, String color) {
042        this.type = type;
043        this.color = color;
044    }
045
046    public ButtonCell(IconType icon, String color) {
047        this.icon = icon;
048        this.color = color;
049    }
050
051    public ButtonCell(IconType icon, ButtonType type) {
052                super();
053                this.icon = icon;
054                this.type = type;
055        }
056
057        @Override
058        public void render(Context context, SafeHtml data, SafeHtmlBuilder sb) {
059                sb.appendHtmlConstant("<button type=\"button\" class=\"btn "
060                                + (type != null ? type.get() : "") + "\" tabindex=\"-1\">");
061                if (data != null) {
062                        if (icon != null) {
063                if(color != null) {
064                    try {
065                        color = " " + IconColor.valueOf(color.toUpperCase()).get();
066                    } catch (IllegalArgumentException e) {
067                        color = "";
068                    }
069                } else {
070                    color = "";
071                }
072
073                sb.appendHtmlConstant("<i class=\"" + icon.get() + color + "\"></i> ");
074            }
075                        sb.append(data);
076                }
077                sb.appendHtmlConstant("</button>");
078        }
079
080    public IconType getIcon() {
081        return icon;
082    }
083
084    public void setIcon(IconType icon) {
085        this.icon = icon;
086    }
087
088    public ButtonType getType() {
089        return type;
090    }
091
092    public void setType(ButtonType type) {
093        this.type = type;
094    }
095
096    public String getColor() {
097        return color;
098    }
099
100    public void setColor(String color) {
101        this.color = color;
102    }
103}
