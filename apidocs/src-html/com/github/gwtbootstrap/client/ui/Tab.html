001package com.github.gwtbootstrap.client.ui;
002
003import java.util.Iterator;
004
005import com.github.gwtbootstrap.client.ui.base.HasIcon;
006import com.github.gwtbootstrap.client.ui.base.HasStyle;
007import com.github.gwtbootstrap.client.ui.base.IsResponsive;
008import com.github.gwtbootstrap.client.ui.base.Style;
009import com.github.gwtbootstrap.client.ui.constants.Device;
010import com.github.gwtbootstrap.client.ui.constants.IconSize;
011import com.github.gwtbootstrap.client.ui.constants.IconType;
012import com.google.gwt.event.dom.client.ClickHandler;
013import com.google.gwt.event.dom.client.HasClickHandlers;
014import com.google.gwt.event.shared.GwtEvent;
015import com.google.gwt.event.shared.HandlerRegistration;
016import com.google.gwt.user.client.DOM;
017import com.google.gwt.user.client.ui.HasWidgets;
018import com.google.gwt.user.client.ui.IsWidget;
019import com.google.gwt.user.client.ui.Widget;
020
021/**
022 * The tab widget for {@link TabPanel}.
023 * 
024 * <p>
025 * It's for UiBinder.
026 * Tab class provide easy syntax on UiBinder.
027 * </p>
028 * Example:
029 * <pre>
030 * {@code
031 * <b:TabPanel>
032 *  <b:Tab heading="Typically">
033 *    <b:Heading size="3">Typically Tab</b:Heading>
034 *    <b:Paragraph>
035 *        huhuhu hahha
036 *    </b:Paragraph>
037 *  </b:Tab>
038 * </b:TabPanel>
039 * }
040 * </pre>
041 * 
042 * 
043 * 
044 * }
045 * 
046 * </pre>
047 * @since 2.0.4.0
048 * @author ohashi keisuke
049 */
050public class Tab implements IsWidget, HasWidgets, HasClickHandlers,HasStyle, IsResponsive,HasIcon {
051    
052    TabLink link = new TabLink();
053
054    /**
055     * Create tmpy tab
056     */
057    public Tab() {
058        TabPane tabPane = new TabPane();
059        
060        tabPane.setHref(DOM.createUniqueId());
061        
062        link.setTablePane(tabPane);
063    }
064    
065    /**
066     * Tab as a TabLink
067     */
068    @Override
069    public Widget asWidget() {
070        return link;
071    }
072    
073    /**
074     * Get Container TabPane
075     * @return TabPane
076     */
077    protected TabPane getTabPane() {
078        return link.getTabPane();
079    }
080    
081    /**
082     * Return TabLink
083     * @return tabLink
084     */
085    public TabLink asTabLink() {
086        return link;
087    }
088    
089    /**
090     * Set tab active
091     * @param active
092     */
093    public void setActive(boolean active) {
094        link.setActive(active);
095    }
096    
097    public boolean isActive() {
098        return link.isActive();
099    }
100    
101    /**
102     * Set tab text
103     * @param text tab text
104     */
105    public void setHeading(String text) {
106        link.setText(text);
107    }
108    
109    /**
110     * Get Tab text
111     * @return tab text
112     */
113    public String getHeading() {
114        return link.getText();
115    }
116    
117    /**
118     * Add widget to tab pane.
119     */
120    @Override
121    public void add(Widget w) {
122        link.getTabPane().add(w);
123    }
124
125    /**
126     * Clear tab pane children
127     */
128    @Override
129    public void clear() {
130        link.getTabPane().clear();
131    }
132
133    @Override
134    public Iterator<Widget> iterator() {
135        return link.getTabPane().iterator();
136    }
137
138    @Override
139    public boolean remove(Widget w) {
140        return link.getTabPane().remove(w);
141    }
142
143    @Override
144    public HandlerRegistration addClickHandler(ClickHandler handler) {
145        return link.addClickHandler(handler);
146    }
147
148    @Override
149    public void setIcon(IconType type) {
150        link.setIcon(type);
151    }
152
153    @Override
154    public void setIconSize(IconSize size) {
155        link.setIconSize(size);
156    }
157
158    @Override
159    public void setShowOn(Device device) {
160        link.setShowOn(device);
161        link.getTabPane().setShowOn(device);
162    }
163
164    @Override
165    public void setHideOn(Device device) {
166        link.setHideOn(device);
167        link.getTabPane().setHideOn(device);
168    }
169
170    @Override
171    public void setStyle(Style style) {
172        link.setStyle(style);
173    }
174
175    @Override
176    public void addStyle(Style style) {
177        link.addStyle(style);
178    }
179
180    @Override
181    public void removeStyle(Style style) {
182        link.removeStyle(style);
183    }
184
185    @Override
186    public void fireEvent(GwtEvent<?> event) {
187        link.fireEvent(event);
188    }
189}
