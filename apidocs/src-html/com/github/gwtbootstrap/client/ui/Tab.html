001package com.github.gwtbootstrap.client.ui;
002
003import java.util.Iterator;
004
005import com.github.gwtbootstrap.client.ui.base.HasIcon;
006import com.github.gwtbootstrap.client.ui.base.HasStyle;
007import com.github.gwtbootstrap.client.ui.base.IsResponsive;
008import com.github.gwtbootstrap.client.ui.base.Style;
009import com.github.gwtbootstrap.client.ui.constants.Device;
010import com.github.gwtbootstrap.client.ui.constants.IconSize;
011import com.github.gwtbootstrap.client.ui.constants.IconType;
012import com.google.gwt.event.dom.client.ClickHandler;
013import com.google.gwt.event.dom.client.HasClickHandlers;
014import com.google.gwt.event.shared.GwtEvent;
015import com.google.gwt.event.shared.HandlerRegistration;
016import com.google.gwt.uibinder.client.UiChild;
017import com.google.gwt.user.client.DOM;
018import com.google.gwt.user.client.ui.HasWidgets;
019import com.google.gwt.user.client.ui.IsWidget;
020import com.google.gwt.user.client.ui.Widget;
021
022/**
023 * The tab widget for {@link TabPanel}.
024 * 
025 * <p>
026 * It's for UiBinder.
027 * Tab class provide easy syntax on UiBinder.
028 * </p>
029 * Example:
030 * <pre>
031 * {@code
032 * <b:TabPanel>
033 *  <b:Tab heading="Typically">
034 *    <b:Heading size="3">Typically Tab</b:Heading>
035 *    <b:Paragraph>
036 *        huhuhu hahha
037 *    </b:Paragraph>
038 *  </b:Tab>
039 * </b:TabPanel>
040 * }
041 * </pre>
042 * 
043 * 
044 * 
045 * }
046 * 
047 * </pre>
048 * @since 2.0.4.0
049 * @author ohashi keisuke
050 */
051public class Tab implements IsWidget, HasWidgets, HasClickHandlers,HasStyle, IsResponsive,HasIcon {
052    
053    TabLink link = new TabLink();
054
055    /**
056     * Create tmpy tab
057     */
058    public Tab() {
059        TabPane tabPane = new TabPane();
060        
061        tabPane.setHref(DOM.createUniqueId());
062        
063        link.setTablePane(tabPane);
064    }
065    
066    /**
067     * Tab as a TabLink
068     */
069    @Override
070    public Widget asWidget() {
071        return link;
072    }
073    
074    /**
075     * Get Container TabPane
076     * @return TabPane
077     */
078    protected TabPane getTabPane() {
079        return link.getTabPane();
080    }
081    
082    /**
083     * Return TabLink
084     * @return tabLink
085     */
086    public TabLink asTabLink() {
087        return link;
088    }
089    
090    /**
091     * Set tab active
092     * @param active
093     */
094    public void setActive(boolean active) {
095        link.setActive(active);
096    }
097    
098    /**
099     * has active style name
100     * @return true:active false:deactive
101     */
102    public boolean isActive() {
103        return link.isActive();
104    }
105    
106    /**
107     * Set tab text
108     * @param text tab text
109     */
110    public void setHeading(String text) {
111        link.setText(text);
112    }
113    
114    /**
115     * Get Tab text
116     * @return tab text
117     */
118    public String getHeading() {
119        return link.getText();
120    }
121    
122    /**
123     * Add widget to tab pane.
124     */
125    @Override
126    public void add(Widget w) {
127        link.getTabPane().add(w);
128    }
129
130    /**
131     * Clear tab pane children
132     */
133    @Override
134    public void clear() {
135        link.getTabPane().clear();
136    }
137
138    /**
139     * call {@link TabPane#iterator()}
140     */
141    @Override
142    public Iterator<Widget> iterator() {
143        return link.getTabPane().iterator();
144    }
145
146    /**
147     * call {@link TabPane#remove(Widget)}
148     * 
149     * @return {@link TabPane#remove(Widget)} result
150     */
151    @Override
152    public boolean remove(Widget w) {
153        return link.getTabPane().remove(w);
154    }
155
156    /**
157     * add ClickEventHandler to TabLink
158     * {@inheritDoc}
159     */
160    @Override
161    public HandlerRegistration addClickHandler(ClickHandler handler) {
162        return link.addClickHandler(handler);
163    }
164
165    /**
166     * set TabLink icon type.
167     * {@inheritDoc}
168     */
169    @Override
170    public void setIcon(IconType type) {
171        link.setIcon(type);
172    }
173
174    /**
175     * Set TabLink icon size
176     * {@inheritDoc}
177     */
178    @Override
179    public void setIconSize(IconSize size) {
180        link.setIconSize(size);
181    }
182
183    /**
184     * Set TabLink and TabPane show on device.
185     * {@inheritDoc}
186     */
187    @Override
188    public void setShowOn(Device device) {
189        link.setShowOn(device);
190        link.getTabPane().setShowOn(device);
191    }
192
193    /**
194     * Set TabLink and TabPane show on device.
195     * {@inheritDoc}
196     */
197    @Override
198    public void setHideOn(Device device) {
199        link.setHideOn(device);
200        link.getTabPane().setHideOn(device);
201    }
202
203    /**
204     * set TabLink style
205     * {@inheritDoc}
206     */
207    @Override
208    public void setStyle(Style style) {
209        link.setStyle(style);
210    }
211
212    /**
213     * add TabLink style
214     * {@inheritDoc}
215     */
216    @Override
217    public void addStyle(Style style) {
218        link.addStyle(style);
219    }
220
221    /**
222     * remove TabLink style
223     * {@inheritDoc}
224     */
225    @Override
226    public void removeStyle(Style style) {
227        link.removeStyle(style);
228    }
229
230    /**
231     * fire TabLink event
232     * {@inheritDoc}
233     */
234    @Override
235    public void fireEvent(GwtEvent<?> event) {
236        link.fireEvent(event);
237    }
238    
239    @UiChild(limit=1,tagname="customTab")
240    public void addDecorate(Widget w) {
241        link.add(w);
242    }
243}
