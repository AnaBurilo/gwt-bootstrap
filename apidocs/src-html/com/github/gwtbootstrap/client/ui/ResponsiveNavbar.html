029    //@formatter:off
030    /**
031     * A {@link Navbar} that hides the contents of a {@link NavCollapse} whenever
032     * the width of the window is too small.
033     * <p>
034     * Only works when the Responsive Layout ist turned on. Create your own
035     * {@link Configurator} and let {@link Configurator#hasResponsiveDesign()}
036     * return true.
037     * 
038     * @since 2.0.2.0
039     * 
040     * @author Dominik Mayer
041     * @author Carlos Alexandro Becker
042     */
043    //@formatter:on
044    public class ResponsiveNavbar extends Navbar {
045    
046            private static final int RESPONSIVE_WIDTH_IN_PIXEL = 980;
047    
048            private final NavbarButton collapseButton = new NavbarButton();
049            private final DivWidget navCollapse = new DivWidget("nav-collapse");
050    
051            public ResponsiveNavbar() {
052                    super();
053                    collapseButton.getElement().setAttribute(Constants.DATA_TOGGLE,
054                                    Bootstrap.collapse);
055                    collapseButton.getElement().setAttribute(Bootstrap.data_target,
056                                    Bootstrap.nav_collapse_target);
057                    collapseButton.add(new Icon(IconType.BAR));
058                    collapseButton.add(new Icon(IconType.BAR));
059                    collapseButton.add(new Icon(IconType.BAR));
060                    add(collapseButton);
061                    addWindowHandlers();
062                    setPaddingTop();
063    
064            }
065    
066            @Override
067            public void add(Widget child) {
068                    if (child instanceof Nav) {
069                            // without this, the order of elements in navbar will bug.
070                            if (!getChildren().contains(navCollapse))
071                                    super.add(navCollapse);
072                            navCollapse.add(child);
073                    } else
074                            super.add(child);
075            }
076    
077            private void addWindowHandlers() {
078    
079                    Window.addResizeHandler(new ResizeHandler() {
080    
081                            Timer resizeTimer = new Timer() {
082    
083                                    @Override
084                                    public void run() {
085                                            setPaddingTop();
086    //                                      navCollapse.getElement().setAttribute("style",
087    //                                                      "height: auto;");
088                                    }
089                            };
090    
091                            public void onResize(final ResizeEvent event) {
092                                    resizeTimer.cancel();
093                                    resizeTimer.schedule(250);
094                            }
095                    });
096            }
097    
098            private void setPaddingTop() {
099                    setPaddingTop(!isResponsive());
100            }
101    
102            private boolean isResponsive() {
103                    return Window.getClientWidth() < RESPONSIVE_WIDTH_IN_PIXEL;
104            }
105    
106            @Override
107            protected Container getContainer() {
108                    return new FluidContainer();
109            }
110    }
