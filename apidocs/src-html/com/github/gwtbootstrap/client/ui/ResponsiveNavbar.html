016    package com.github.gwtbootstrap.client.ui;
017    
018    import com.github.gwtbootstrap.client.ui.base.DivWidget;
019    import com.github.gwtbootstrap.client.ui.constants.Constants;
020    import com.github.gwtbootstrap.client.ui.resources.Bootstrap;
021    import com.github.gwtbootstrap.client.ui.resources.ResourceAdapter;
022    import com.google.gwt.event.logical.shared.ResizeEvent;
023    import com.google.gwt.event.logical.shared.ResizeHandler;
024    import com.google.gwt.user.client.Timer;
025    import com.google.gwt.user.client.Window;
026    import com.google.gwt.user.client.ui.Widget;
027    
028    /**
029     * A {@link Navbar} that hides the contents of a {@link NavCollapse} whenever
030     * the width of the window is too small.
031     * <p>
032     * Only works when the Responsive Layout ist turned on. Create your own
033     * {@link ResourceAdapter} and let {@link ResourceAdapter#hasResponsiveDesign()}
034     * return true.
035     * 
036     * @since 2.0.2.0
037     * 
038     * @author Dominik Mayer
039     * @author Carlos Alexandro Becker
040     */
041    public class ResponsiveNavbar extends Navbar {
042    
043            private static final int RESPONSIVE_WIDTH_IN_PIXEL = 980;
044    
045            private final NavbarButton collapseButton = new NavbarButton();
046            private final DivWidget navCollapse = new DivWidget("nav-collapse");
047    
048            public ResponsiveNavbar() {
049                    super();
050                    collapseButton.getElement().setAttribute(Constants.DATA_TOGGLE,
051                                    Bootstrap.collapse);
052                    collapseButton.getElement().setAttribute(Bootstrap.data_target,
053                                    Bootstrap.nav_collapse_target);
054                    collapseButton.add(new Icon("bar"));
055                    collapseButton.add(new Icon("bar"));
056                    collapseButton.add(new Icon("bar"));
057                    add(collapseButton);
058                    addWindowHandlers();
059                    setPaddingTop();
060    
061            }
062    
063            @Override
064            public void add(Widget child) {
065                    if (child instanceof Nav) {
066                            // without this, the order of elements in navbar will bug.
067                            if (!getChildren().contains(navCollapse))
068                                    super.add(navCollapse);
069                            navCollapse.add(child);
070                    } else
071                            super.add(child);
072            }
073    
074            private void addWindowHandlers() {
075    
076                    Window.addResizeHandler(new ResizeHandler() {
077    
078                            Timer resizeTimer = new Timer() {
079    
080                                    @Override
081                                    public void run() {
082                                            setPaddingTop();
083    //                                      navCollapse.getElement().setAttribute("style",
084    //                                                      "height: auto;");
085                                    }
086                            };
087    
088                            public void onResize(final ResizeEvent event) {
089                                    resizeTimer.cancel();
090                                    resizeTimer.schedule(250);
091                            }
092                    });
093            }
094    
095            private void setPaddingTop() {
096                    setPaddingTop(!isResponsive());
097            }
098    
099            private boolean isResponsive() {
100                    return Window.getClientWidth() < RESPONSIVE_WIDTH_IN_PIXEL;
101            }
102    
103            @Override
104            protected Container getContainer() {
105                    return new FluidContainer();
106            }
107    }
