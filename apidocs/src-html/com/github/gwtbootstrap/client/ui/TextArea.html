001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.TextBoxBase;
004import com.google.gwt.dom.client.Document;
005import com.google.gwt.dom.client.Element;
006import com.google.gwt.dom.client.TextAreaElement;
007import com.google.gwt.user.client.ui.RootPanel;
008import com.google.gwt.user.client.ui.Widget;
009
010/**
011 * A TextArea for Bootstrap form.
012 * 
013 * @since 2.0.3.0
014 * @author ohashi keisuke
015 */
016public class TextArea extends TextBoxBase {
017
018        /**
019         * Creates a TextArea widget that wraps an existing &lt;textarea&gt;
020         * element.
021         * 
022         * This element must already be attached to the document. If the element is
023         * removed from the document, you must call
024         * {@link RootPanel#detachNow(Widget)}.
025         * 
026         * @param element
027         *            the element to be wrapped
028         */
029        public static TextArea wrap(Element element) {
030                // Assert that the element is attached.
031                assert Document.get().getBody().isOrHasChild(element);
032
033                TextArea textArea = new TextArea(element);
034
035                // Mark it attached and remember it for cleanup.
036                textArea.onAttach();
037                RootPanel.detachOnWindowClose(textArea);
038
039                return textArea;
040        }
041
042        /**
043         * Creates an empty text area.
044         */
045        public TextArea() {
046                super(Document.get().createTextAreaElement());
047                setStyleName("gwt-TextArea");
048        }
049
050        /**
051         * This constructor may be used by subclasses to explicitly use an existing
052         * element. This element must be a &lt;textarea&gt; element.
053         * 
054         * @param element
055         *            the element to be used
056         */
057        protected TextArea(Element element) {
058                super(element.<Element> cast());
059                TextAreaElement.as(element);
060        }
061
062        /**
063         * Gets the requested width of the text box (this is not an exact value, as
064         * not all characters are created equal).
065         * 
066         * @return the requested width, in characters
067         */
068        public int getCharacterWidth() {
069                return getTextAreaElement().getCols();
070        }
071
072        @Override
073        public int getCursorPos() {
074                return getImpl().getTextAreaCursorPos(getElement());
075        }
076
077        @Override
078        public int getSelectionLength() {
079                return getImpl().getTextAreaSelectionLength(getElement());
080        }
081
082        /**
083         * Gets the number of text lines that are visible.
084         * 
085         * @return the number of visible lines
086         */
087        public int getVisibleLines() {
088                return getTextAreaElement().getRows();
089        }
090
091        /**
092         * Sets the requested width of the text box (this is not an exact value, as
093         * not all characters are created equal).
094         * 
095         * @param width
096         *            the requested width, in characters
097         */
098        public void setCharacterWidth(int width) {
099                getTextAreaElement().setCols(width);
100        }
101
102        /**
103         * Sets the number of text lines that are visible.
104         * 
105         * @param lines
106         *            the number of visible lines
107         */
108        public void setVisibleLines(int lines) {
109                getTextAreaElement().setRows(lines);
110        }
111
112        private TextAreaElement getTextAreaElement() {
113                return getElement().cast();
114        }
115
116}
