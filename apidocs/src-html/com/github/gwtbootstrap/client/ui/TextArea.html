001package com.github.gwtbootstrap.client.ui;
002
003import com.github.gwtbootstrap.client.ui.base.HasAlternateSize;
004import com.github.gwtbootstrap.client.ui.base.HasSize;
005import com.github.gwtbootstrap.client.ui.base.SizeHelper;
006import com.github.gwtbootstrap.client.ui.base.StyleHelper;
007import com.github.gwtbootstrap.client.ui.constants.AlternateSize;
008import com.google.gwt.dom.client.Document;
009import com.google.gwt.dom.client.Element;
010import com.google.gwt.dom.client.TextAreaElement;
011
012public class TextArea extends com.google.gwt.user.client.ui.TextArea implements HasSize , HasAlternateSize {
013
014        /**
015         * Creates an empty text area.
016         */
017        public TextArea() {
018                super(Document.get().createTextAreaElement());
019        }
020
021        /**
022         * This constructor may be used by subclasses to explicitly use an existing
023         * element. This element must be a &lt;textarea&gt; element.
024         * 
025         * @param element
026         *            the element to be used
027         */
028        protected TextArea(Element element) {
029                super(element.<Element> cast());
030                TextAreaElement.as(element);
031        }
032
033        /**
034         * {@inheritDoc}
035         */
036        @Override
037        public void setAlternateSize(AlternateSize size) {
038                StyleHelper.changeStyle(this, size, AlternateSize.class);
039        }
040
041        /**
042         * {@inheritDoc}
043         */
044        @Override
045        public void setSize(int size) {
046                SizeHelper.setSize(this, size);
047        }
048        
049        
050}
