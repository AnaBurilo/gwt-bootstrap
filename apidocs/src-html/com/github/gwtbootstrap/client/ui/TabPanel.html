016package com.github.gwtbootstrap.client.ui;
017
018import com.github.gwtbootstrap.client.ui.base.DivWidget;
019import com.github.gwtbootstrap.client.ui.base.IconAnchor;
020import com.github.gwtbootstrap.client.ui.constants.Constants;
021import com.github.gwtbootstrap.client.ui.resources.Bootstrap;
022import com.google.gwt.core.client.GWT;
023import com.google.gwt.user.client.Element;
024import com.google.gwt.user.client.ui.Widget;
025
026//@formatter:off
027/**
028 * The container for a tabbable nav.
029 * 
030 * @since 2.0.4.0
031 * @author Dominik Mayer
032 */
033//@formatter:on
034public class TabPanel extends DivWidget {
035
036        private static class TabContent extends DivWidget {
037
038                public TabContent() {
039                        setStyleName(Bootstrap.tab_content);
040                }
041        }
042
043        static class TabLink extends NavLink {
044
045                public TabLink(TabPane pane) {
046                        super(pane.getHeading());
047                        GWT.log("create tab link");
048
049                        IconAnchor anchor = getAnchor();
050                        anchor.getElement().setAttribute(Constants.DATA_TOGGLE, "tab");
051                        anchor.getElement().setAttribute(Constants.DATA_TARGET,  "#" + pane.getId());
052                        if (pane.isActive()) {
053                                GWT.log("set active to tab link");
054                                show();
055                        }
056                }
057                
058                public void show() {
059                        setActive(true);
060                        show(getAnchor().getElement());
061                }
062                
063                public native void show(Element e)/*-{
064                        $wnd.jQuery(e).tab('show');
065                }-*/;
066
067        }
068
069        private NavTabs tabs = new NavTabs();
070
071        private TabContent tabContent = new TabContent();
072
073        public TabPanel() {
074                this(Bootstrap.Tabs.ABOVE);
075        }
076
077        public TabPanel(Bootstrap.Tabs position) {
078                setStyle(position);
079                super.add(tabs);
080                super.add(tabContent);
081        }
082
083        public void setTabPosition(String position) {
084                if (position.equalsIgnoreCase("below"))
085                        setStyle(Bootstrap.Tabs.BELOW);
086                else if (position.equalsIgnoreCase("left"))
087                        setStyle(Bootstrap.Tabs.LEFT);
088                else if (position.equalsIgnoreCase("right"))
089                        setStyle(Bootstrap.Tabs.RIGHT);
090                else
091                        setStyle(Bootstrap.Tabs.ABOVE);
092        }
093
094        @Override
095        public void add(Widget child) {
096
097                if (!(child instanceof TabPane))
098                        throw new IllegalArgumentException("Only Tab Panes can be added"
099                                        + "to a TabPanel");
100
101                add((TabPane) child);
102        }
103
104        private void add(TabPane child) {
105
106                TabLink tabLink = new TabLink(child);
107                tabs.add(tabLink);
108                tabContent.add(child);
109        }
110        
111        @Override
112        protected void onAttach() {
113                super.onAttach();
114                
115                for(int i = 0;i < tabs.getWidgetCount();i++) {
116                        
117                        Widget widget = tabs.getWidget(i);
118                        
119                        if (widget instanceof TabLink) {
120                                TabLink tabLink = (TabLink) widget;
121                                
122                                if(tabLink.isActive()) {
123                                        tabLink.show();
124                                        break;
125                                }
126                        }
127                }
128        }
129}
