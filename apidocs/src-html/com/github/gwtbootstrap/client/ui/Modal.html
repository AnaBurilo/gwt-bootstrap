042    //@formatter:off
043    /**
044     * Popup dialog with optional header and {@link ModalFooter footer.}
045     * <p>
046     * By default, all other Modals are closed once a new one is opened. This
047     * setting can be {@link #setHideOthers(boolean) overridden.}
048     * 
049     * <p>
050     * <h3>UiBinder Usage:</h3>
051     * 
052     * <pre>
053     * {@code
054     * <b:Modal title="My Modal" backdrop="STATIC">
055     *     <g:Label>Modal Content!</g:Label>
056     *     <b:ModalFooter>
057     *         <b:Button icon="FILE">Save</b:Button>
058     *     </b:ModalFooter>
059     * </b:Modal>
060     * }
061     * </pre>
062     * All arguments are optional.
063     * </p>
064     * 
065     * @since 2.0.2.0
066     * 
067     * @author Carlos Alexandro Becker
068     * 
069     * @author Dominik Mayer
070     * 
071     * @see <a href="http://twitter.github.com/bootstrap/javascript.html#modals">Bootstrap documentation</a>
072     * @see PopupPanel
073     */
074    //@formatter:on
075    public class Modal extends DivWidget implements HasVisibility,
076                    HasVisibleHandlers, IsAnimated {
077    
078            private static Set<Modal> currentlyShown = new HashSet<Modal>();
079    
080            private final DivWidget header = new DivWidget();
081            private final DivWidget body = new DivWidget("modal-body");
082    
083            private boolean keyboard = true;
084            private BackdropType backdropType = BackdropType.NORMAL;
085            private boolean show = false;
086            private boolean hideOthers = true;
087    
088            private boolean configured = false;
089    
090            /**
091             * Creates an empty, hidden widget.
092             */
093            public Modal() {
094                    super("modal");
095                    super.add(header);
096                    super.add(body);
097                    setVisible(false);
098                    RootPanel.get().add(this);
099                    setHandlerFunctions(getElement());
100            }
101    
102            /**
103             * Creates an empty, hidden widget with specified show behavior.
104             * 
105             * @param animated
106             *            <code>true</code> if the widget should be animated.
107             */
108            public Modal(boolean animated) {
109                    this();
110                    setAnimation(animated);
111            }
112    
113            /**
114             * Sets the title of the Modal.
115             * 
116             * @param title
117             *            the title of the Modal
118             */
119            @Override
120            public void setTitle(String title) {
121                    header.clear();
122                    if (title == null || title.isEmpty()) {
123                            showHeader(false);
124                    } else {
125                            header.add(new Close(DismissType.MODAL));
126                            header.add(new Heading(3, title));
127                            showHeader(true);
128                    }
129            }
130    
131            private void showHeader(boolean show) {
132                    if (show)
133                            header.setStyleName(Constants.MODAL_HEADER);
134                    else
135                            header.removeStyleName(Constants.MODAL_HEADER);
136            }
137    
138            /**
139             * {@inheritDoc}
140             */
141            public void setAnimation(boolean animated) {
142                    if (animated)
143                            addStyleName(Constants.FADE);
144                    else
145                            removeStyleName(Constants.FADE);
146            }
147    
148            /**
149             * {@inheritDoc}
150             */
151            public boolean getAnimation() {
152                    return getStyleName().contains(Constants.FADE);
153            }
154    
155            /**
156             * Sets whether this Modal appears on top of others or is the only one
157             * visible on screen.
158             * 
159             * @param hideOthers
160             *            <code>true</code> to make sure that this modal is the only one
161             *            shown. All others will be hidden. Default: <code>true</code>
162             */
163            public void setHideOthers(boolean hideOthers) {
164                    this.hideOthers = hideOthers;
165            }
166    
167            /**
168             * Sets whether the Modal is closed when the <code>ESC</code> is pressed.
169             * 
170             * @param keyboard
171             *            <code>true</code> if the Modal is closed by <code>ESC</code>
172             *            key. Default: <code>true</code>
173             */
174            public void setKeyboard(boolean keyboard) {
175                    this.keyboard = keyboard;
176                    reconfigure();
177            }
178    
179            /**
180             * Sets the type of the backdrop.
181             * 
182             * @param type
183             *            the backdrop type
184             */
185            public void setBackdrop(BackdropType type) {
186                    backdropType = type;
187                    reconfigure();
188            }
189    
190            /**
191             * Reconfigures the modal with changed settings.
192             */
193            protected void reconfigure() {
194                    if (configured) {
195                            configure(keyboard, backdropType, show);
196                    }
197            }
198    
199            /**
200             * {@inheritDoc}
201             */
202            @Override
203            public void add(Widget w) {
204                    if (w instanceof ModalFooter) {
205                            super.add(w);
206                    } else
207                            body.add(w);
208            }
209    
210            /**
211             * {@inheritDoc}
212             */
213            @Override
214            public void insert(Widget w, int beforeIndex) {
215                    body.insert(w, beforeIndex);
216            }
217    
218            /**
219             * {@inheritDoc}
220             */
221            @Override
222            protected void onLoad() {
223                    super.onLoad();
224                    configure(keyboard, backdropType, show);
225                    configured = true;
226            }
227    
228            /**
229             * {@inheritDoc}
230             */
231            public void show() {
232                    changeVisibility("show");
233            }
234    
235            /**
236             * {@inheritDoc}
237             */
238            public void hide() {
239                    changeVisibility("hide");
240            }
241    
242            /**
243             * {@inheritDoc}
244             */
245            public void toggle() {
246                    changeVisibility("toggle");
247            }
248    
249            private void changeVisibility(String visibility) {
250                    changeVisibility(getElement(), visibility);
251            }
252    
253            /**
254             * This method is called immediately when the widget's {@link #hide()}
255             * method is executed.
256             */
257            protected void onHide() {
258                    fireEvent(new HideEvent());
259            }
260    
261            /**
262             * This method is called once the widget is completely hidden.
263             */
264            protected void onHidden() {
265                    fireEvent(new HiddenEvent());
266                    currentlyShown.remove(this);
267            }
268    
269            /**
270             * This method is called immediately when the widget's {@link #show()}
271             * method is executed.
272             */
273            protected void onShow() {
274                    if (hideOthers)
275                            hideShownModals();
276                    fireEvent(new ShowEvent());
277            }
278    
279            private void hideShownModals() {
280                    for (Modal m : currentlyShown)
281                            m.hide();
282            }
283    
284            /**
285             * This method is called once the widget is completely shown.
286             */
287            protected void onShown() {
288                    fireEvent(new ShownEvent());
289                    currentlyShown.add(this);
290            }
291    
292            private void configure(boolean keyboard, BackdropType backdropType,
293                            boolean show) {
294                    if (backdropType == BackdropType.NORMAL)
295                            configure(getElement(), keyboard, true, show);
296                    else if (backdropType == BackdropType.NONE)
297                            configure(getElement(), keyboard, false, show);
298                    else if (backdropType == BackdropType.STATIC)
299                            configure(getElement(), keyboard, BackdropType.STATIC.get(), show);
300            }
301    
302            private native void configure(Element e, boolean k, boolean b, boolean s) /*-{
303                    $wnd.jQuery(e).modal({
304                            keyboard : k,
305                            backdrop : b,
306                            show : s
307                    });
308            }-*/;
309    
310            private native void configure(Element e, boolean k, String b, boolean s) /*-{
311                    $wnd.jQuery(e).modal({
312                            keyboard : k,
313                            backdrop : b,
314                            show : s
315                    });
316            }-*/;
317    
318            private native void changeVisibility(Element e, String visibility) /*-{
319                    $wnd.jQuery(e).modal(visibility);
320            }-*/;
321    
322            /**
323             * Links the Java functions that fire the events.
324             */
325            private native void setHandlerFunctions(Element e) /*-{
326                    var that = this;
327                    $wnd.jQuery(e).on('hide', function() {
328                            that.@com.github.gwtbootstrap.client.ui.Modal::onHide()();
329                    });
330                    $wnd.jQuery(e).on('hidden', function() {
331                            that.@com.github.gwtbootstrap.client.ui.Modal::onHidden()();
332                    });
333                    $wnd.jQuery(e).on('show', function() {
334                            that.@com.github.gwtbootstrap.client.ui.Modal::onShow()();
335                    });
336                    $wnd.jQuery(e).on('shown', function() {
337                            that.@com.github.gwtbootstrap.client.ui.Modal::onShown()();
338                    });
339            }-*/;
340    
341            /**
342             * {@inheritDoc}
343             */
344            public HandlerRegistration addHideHandler(HideHandler handler) {
345                    return addHandler(handler, HideEvent.getType());
346            }
347    
348            /**
349             * {@inheritDoc}
350             */
351            public HandlerRegistration addHiddenHandler(HiddenHandler handler) {
352                    return addHandler(handler, HiddenEvent.getType());
353            }
354    
355            /**
356             * {@inheritDoc}
357             */
358            public HandlerRegistration addShowHandler(ShowHandler handler) {
359                    return addHandler(handler, ShowEvent.getType());
360            }
361    
362            /**
363             * {@inheritDoc}
364             */
365            public HandlerRegistration addShownHandler(ShownHandler handler) {
366                    return addHandler(handler, ShownEvent.getType());
367            }
