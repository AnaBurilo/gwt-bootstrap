016    package com.github.gwtbootstrap.client.ui;
017    
018    import java.util.HashSet;
019    import java.util.Set;
020    
021    import com.github.gwtbootstrap.client.ui.base.DivWidget;
022    import com.github.gwtbootstrap.client.ui.base.HasVisibility;
023    import com.github.gwtbootstrap.client.ui.base.HasVisibleHandlers;
024    import com.github.gwtbootstrap.client.ui.base.IsAnimated;
025    import com.github.gwtbootstrap.client.ui.constants.DismissType;
026    import com.github.gwtbootstrap.client.ui.event.HiddenEvent;
027    import com.github.gwtbootstrap.client.ui.event.HiddenHandler;
028    import com.github.gwtbootstrap.client.ui.event.HideEvent;
029    import com.github.gwtbootstrap.client.ui.event.HideHandler;
030    import com.github.gwtbootstrap.client.ui.event.ShowEvent;
031    import com.github.gwtbootstrap.client.ui.event.ShowHandler;
032    import com.github.gwtbootstrap.client.ui.event.ShownEvent;
033    import com.github.gwtbootstrap.client.ui.event.ShownHandler;
034    import com.google.gwt.dom.client.Element;
035    import com.google.gwt.event.shared.HandlerRegistration;
036    import com.google.gwt.user.client.ui.RootPanel;
037    import com.google.gwt.user.client.ui.Widget;
038    
039    /**
040     * 
041     * 
042     * @since 2.0.2.0
043     * 
044     * @author Carlos Alexandro Becker
045     * 
046     * @author Dominik Mayer
047     */
048    public class Modal extends DivWidget implements HasVisibility,
049                    HasVisibleHandlers, IsAnimated {
050    
051            private static Set<Modal> currentlyShown = new HashSet<Modal>();
052    
053            private final DivWidget header = new DivWidget();
054            private final DivWidget body = new DivWidget("modal-body");
055    
056            private boolean keyboard = true;
057            private boolean backdrop = true;
058            private boolean show = false;
059            private boolean hideOthers = true;
060    
061            private boolean configured = false;
062    
063            public Modal() {
064                    super("modal");
065                    super.add(header);
066                    super.add(body);
067                    setVisible(false);
068                    RootPanel.get().add(this);
069                    setHandlerFunctions(getElement());
070            }
071    
072            public Modal(boolean animated) {
073                    this();
074                    setAnimation(animated);
075            }
076    
077            @Override
078            public void setTitle(String title) {
079                    header.clear();
080                    if (title == null || title.isEmpty()) {
081                            showHeader(false);
082                    } else {
083                            header.add(new Close(DismissType.MODAL));
084                            header.add(new Heading(title, 3));
085                            showHeader(true);
086                    }
087            }
088    
089            private void showHeader(boolean show) {
090                    if (show)
091                            header.setStyleName("modal-header");
092                    else
093                            header.removeStyleName("modal-header");
094            }
095    
096            public void setAnimation(boolean animated) {
097                    if (animated)
098                            addStyleName("fade");
099                    else
100                            removeStyleName("fade");
101            }
102    
103            public boolean getAnimation() {
104                    return getStyleName().contains("fade");
105            }
106    
107            /**
108             * Sets whether this Modal appears on top of others or is the only one
109             * visible on screen.
110             * 
111             * @param hideOthers
112             *            <code>true</code> to make sure that this modal is the only one
113             *            shown. All others will be hidden. Default: <code>true</code>
114             */
115            public void setHideOthers(boolean hideOthers) {
116                    this.hideOthers = hideOthers;
117            }
118    
119            public void setShow(boolean show) {
120                    this.show = show;
121                    reconfigure();
122            }
123    
124            public void setKeyboard(boolean keyboard) {
125                    this.keyboard = keyboard;
126                    reconfigure();
127            }
128    
129            public void setBackdrop(boolean backdrop) {
130                    this.backdrop = backdrop;
131                    reconfigure();
132            }
133    
134            public void reconfigure() {
135                    if (configured) {
136                            configure(getElement(), keyboard, backdrop, show);
137                    }
138            }
139    
140            @Override
141            public void add(Widget w) {
142                    if (w instanceof ModalFooter) {
143                            super.add(w);
144                    } else
145                            body.add(w);
146            }
147    
148            @Override
149            public void insert(Widget w, int beforeIndex) {
150                    body.insert(w, beforeIndex);
151            }
152    
153            @Override
154            protected void onLoad() {
155                    super.onLoad();
156                    configure(getElement(), keyboard, backdrop, show);
157                    configured = true;
158            }
159    
160            /**
161             * {@inheritDoc}
162             */
163            public void show() {
164                    changeVisibility("show");
165            }
166    
167            /**
168             * {@inheritDoc}
169             */
170            public void hide() {
171                    changeVisibility("hide");
172            }
173    
174            /**
175             * {@inheritDoc}
176             */
177            public void toggle() {
178                    changeVisibility("toggle");
179            }
180    
181            private void changeVisibility(String visibility) {
182                    changeVisibility(getElement(), visibility);
183            }
184    
185            /**
186             * {@inheritDoc}
187             */
188            public HandlerRegistration addHideHandler(HideHandler handler) {
189                    return addHandler(handler, HideEvent.getType());
190            }
191    
192            /**
193             * {@inheritDoc}
194             */
195            public HandlerRegistration addHiddenHandler(HiddenHandler handler) {
196                    return addHandler(handler, HiddenEvent.getType());
197            }
198    
199            /**
200             * {@inheritDoc}
201             */
202            public HandlerRegistration addShowHandler(ShowHandler handler) {
203                    return addHandler(handler, ShowEvent.getType());
204            }
205    
206            /**
207             * {@inheritDoc}
208             */
209            public HandlerRegistration addShownHandler(ShownHandler handler) {
210                    return addHandler(handler, ShownEvent.getType());
211            }
212    
213            /**
214             * This method is called immediately when the widget's {@link #hide()}
215             * method is executed.
216             */
217            protected void onHide() {
218                    fireEvent(new HideEvent());
219            }
220    
221            /**
222             * This method is called once the widget is completely hidden.
223             */
224            protected void onHidden() {
225                    fireEvent(new HiddenEvent());
226                    currentlyShown.remove(this);
227            }
228    
229            /**
230             * This method is called immediately when the widget's {@link #show()}
231             * method is executed.
232             */
233            protected void onShow() {
234                    if (hideOthers)
235                            hideShownModals();
236                    fireEvent(new ShowEvent());
237            }
238    
239            private void hideShownModals() {
240                    for (Modal m : currentlyShown)
241                            m.hide();
242            }
243    
244            /**
245             * This method is called once the widget is completely shown.
246             */
247            protected void onShown() {
248                    fireEvent(new ShownEvent());
249                    currentlyShown.add(this);
250            }
251    
252            /*
253             * native functions
254             */
255            private native void configure(Element e, boolean k, boolean b, boolean s) /*-{
256                    $wnd.jQuery(e).modal({
257                            keyboard : k,
258                            backdrop : b,
259                            show : s
260                    });
261            }-*/;
262    
263            private native void changeVisibility(Element e, String visibility) /*-{
264                    $wnd.jQuery(e).modal(visibility);
265            }-*/;
266    
267            /**
268             * Links the Java functions that fire the events.
269             */
270            private native void setHandlerFunctions(Element e) /*-{
271                    var that = this;
272                    $wnd.jQuery(e).on('hide', function() {
273                            that.@com.github.gwtbootstrap.client.ui.Modal::onHide()();
274                    });
275                    $wnd.jQuery(e).on('hidden', function() {
276                            that.@com.github.gwtbootstrap.client.ui.Modal::onHidden()();
277                    });
278                    $wnd.jQuery(e).on('show', function() {
279                            that.@com.github.gwtbootstrap.client.ui.Modal::onShow()();
280                    });
281                    $wnd.jQuery(e).on('shown', function() {
282                            that.@com.github.gwtbootstrap.client.ui.Modal::onShown()();
283                    });
284            }-*/;
285    }
